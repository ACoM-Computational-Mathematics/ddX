192subroutine cosmo_solvation_force_terms(params, constants, workspace, &
193 & state, e_cav, force, error)
194 implicit none
195 type(ddx_params_type), intent(in) :: params
196 type(ddx_constants_type), intent(in) :: constants
197 type(ddx_workspace_type), intent(inout) :: workspace
198 type(ddx_state_type), intent(inout) :: state
199 type(ddx_error_type), intent(inout) :: error
200 real(dp), intent(in) :: e_cav(3, constants % ncav)
201 real(dp), intent(inout) :: force(3, params % nsph)
202 ! local variables
203 real(dp) :: start_time, finish_time
204 integer :: isph
205
206 start_time = omp_get_wtime()
207
208 force = zero
209 do isph = 1, params % nsph
210 call contract_grad_l(params, constants, isph, state % xs, &
211 & state % sgrid, workspace % tmp_vylm(:, 1), &
212 & workspace % tmp_vdylm(:, :, 1), workspace % tmp_vplm(:, 1), &
213 & workspace % tmp_vcos(:, 1), workspace % tmp_vsin(:, 1), &
214 & force(:, isph))
215 call contract_grad_u(params, constants, isph, state % sgrid, &
216 & state % phi_grid, force(:, isph))
217 end do
218
219 force = -pt5*force
220
221 call zeta_grad(params, constants, state, e_cav, force)
222
223 finish_time = omp_get_wtime()
224 state % force_time = finish_time - start_time

236subroutine cosmo_derivative_setup(params, constants, workspace, state)
237 type(ddx_params_type), intent(in) :: params
238 type(ddx_constants_type), intent(in) :: constants
239 type(ddx_workspace_type), intent(inout) :: workspace
240 type(ddx_state_type), intent(inout) :: state
241
242 real(dp), external :: ddot
243 integer :: icav, isph, igrid
244
245 ! Get values of S on the grid
246 call ddeval_grid_work(constants % nbasis, params % ngrid, params % nsph, &
247 & constants % vgrid, constants % vgrid_nbasis, one, state % s, zero, &
248 & state % sgrid)
249 ! Get the values of phi on the grid
250 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
251 & constants % icav_ia, constants % icav_ja, state % phi_cav, &
252 & state % phi_grid)
253
254 ! assemble the intermediate zeta: S weighted by U evaluated on the
255 ! exposed grid points.
256 icav = 0
257 do isph = 1, params % nsph
258 do igrid = 1, params % ngrid
259 if (constants % ui(igrid, isph) .ne. zero) then
260 icav = icav + 1
261 state % zeta(icav) = constants % wgrid(igrid) * &
262 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
263 & constants % vgrid(1, igrid), 1, state % s(1, isph), 1)
264 end if
265 end do
266 end do
267
268end subroutine cosmo_derivative_setup
