268subroutine ddcosmo_solvation_force_terms(params, constants, workspace, &
269 & state, e_cav, force, error)
270 implicit none
271 type(ddx_params_type), intent(in) :: params
272 type(ddx_constants_type), intent(in) :: constants
273 type(ddx_workspace_type), intent(inout) :: workspace
274 type(ddx_state_type), intent(inout) :: state
275 type(ddx_error_type), intent(inout) :: error
276 real(dp), intent(in) :: e_cav(3, constants % ncav)

312subroutine ddcosmo_derivative_setup(params, constants, workspace, state)
313 type(ddx_params_type), intent(in) :: params
314 type(ddx_constants_type), intent(in) :: constants
315 type(ddx_workspace_type), intent(inout) :: workspace
316 type(ddx_state_type), intent(inout) :: state
317
318 real(dp), external :: ddot
319 integer :: icav, isph, igrid
320
321 ! Get values of S on the grid
322 call ddeval_grid_work(constants % nbasis, params % ngrid, params % nsph, &
323 & constants % vgrid, constants % vgrid_nbasis, one, state % s, zero, &
324 & state % sgrid)
325 ! Get the values of phi on the grid
326 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
327 & constants % icav_ia, constants % icav_ja, state % phi_cav, &
328 & state % phi_grid)
329
330 ! assemble the intermediate zeta: S weighted by U evaluated on the
331 ! exposed grid points.
332 icav = 0
333 do isph = 1, params % nsph
334 do igrid = 1, params % ngrid
335 if (constants % ui(igrid, isph) .ne. zero) then
336 icav = icav + 1
337 state % zeta(icav) = constants % wgrid(igrid) * &
338 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
339 & constants % vgrid(1, igrid), 1, state % s(1, isph), 1)
340 end if
341 end do
