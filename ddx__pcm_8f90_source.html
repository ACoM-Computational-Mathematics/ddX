176subroutine pcm_solve_adjoint(params, constants, workspace, state, tol, error)
177 implicit none
178 type(ddx_params_type), intent(in) :: params
179 type(ddx_constants_type), intent(in) :: constants
180 type(ddx_workspace_type), intent(inout) :: workspace
181 type(ddx_state_type), intent(inout) :: state
182 real(dp), intent(in) :: tol
183 type(ddx_error_type), intent(inout) :: error
184 ! local variables
185 real(dp) :: start_time, finish_time
186
187 state % s_niter = params % maxiter
188 start_time = omp_get_wtime()
189 call jacobi_diis(params, constants, workspace, tol, state % psi, &
190 & state % s, state % s_niter, state % s_rel_diff, lstarx, ldm1x, &
191 & hnorm, error)

202 start_time = omp_get_wtime()
203 call jacobi_diis(params, constants, workspace, tol, state % s, state % y, &

212 return
213 end if

215 ! compute the real adjoint solution and store it in Q
216 state % q = state % s - fourpi/(params % eps - one)*state % y

233subroutine pcm_solvation_force_terms(params, constants, workspace, &
234 & state, e_cav, force, error)
235 implicit none
236 type(ddx_params_type), intent(in) :: params
237 type(ddx_constants_type), intent(in) :: constants
238 type(ddx_workspace_type), intent(inout) :: workspace
239 type(ddx_state_type), intent(inout) :: state
240 type(ddx_error_type), intent(inout) :: error
241 real(dp), intent(in) :: e_cav(3, constants % ncav)
242 real(dp), intent(out) :: force(3, params % nsph)
243
244 real(dp) :: start_time, finish_time
245 integer :: isph
246
247 start_time = omp_get_wtime()
248
249 call gradr(params, constants, workspace, state % g, state % ygrid, force)
250
251 do isph = 1, params % nsph
252 call contract_grad_l(params, constants, isph, state % xs, &
253 & state % sgrid, workspace % tmp_vylm(:, 1), &
254 & workspace % tmp_vdylm(:, :, 1), workspace % tmp_vplm(:, 1), &
255 & workspace % tmp_vcos(:, 1), workspace % tmp_vsin(:, 1), &
256 & force(:, isph))
257 call contract_grad_u(params, constants, isph, state % qgrid, &
258 & state % phi_grid, force(:, isph))
259 end do
260 force = - pt5 * force
261
262 call zeta_grad(params, constants, state, e_cav, force)
263
264 finish_time = omp_get_wtime()
265 state % force_time = finish_time - start_time

277subroutine pcm_derivative_setup(params, constants, workspace, state)
278 type(ddx_params_type), intent(in) :: params
279 type(ddx_constants_type), intent(in) :: constants
280 type(ddx_workspace_type), intent(inout) :: workspace
281 type(ddx_state_type), intent(inout) :: state
282
283 real(dp), external :: ddot
284 integer :: icav, isph, igrid
285
286 ! Get grid values of S and Y
287 call dgemm('T', 'N', params % ngrid, params % nsph, &
288 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
289 & state % s, constants % nbasis, zero, state % sgrid, params % ngrid)
290 call dgemm('T', 'N', params % ngrid, params % nsph, &
291 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
292 & state % y, constants % nbasis, zero, state % ygrid, params % ngrid)
293
294 ! Get the values of phi on the grid
295 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
296 & constants % icav_ia, constants % icav_ja, state % phi_cav, &
297 & state % phi_grid)
298
299 state % g = - (state % phieps - state % phi)
300 state % qgrid = state % sgrid - fourpi/(params % eps - one)*state % ygrid
301
302 ! assemble the intermediate zeta: S weighted by U evaluated on the
303 ! exposed grid points.
304 icav = 0
305 do isph = 1, params % nsph
306 do igrid = 1, params % ngrid
307 if (constants % ui(igrid, isph) .ne. zero) then
308 icav = icav + 1
309 state % zeta(icav) = constants % wgrid(igrid) * &
310 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
311 & constants % vgrid(1, igrid), 1, state % q(1, isph), 1)
312 end if
313 end do
314 end do
315
316end subroutine pcm_derivative_setup
