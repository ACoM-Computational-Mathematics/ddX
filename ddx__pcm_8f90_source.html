40subroutine ddpcm(params, constants, workspace, state, phi_cav, &
41 & psi, tol, esolv, force, error)
42 implicit none
43 type(ddx_params_type), intent(in) :: params
44 type(ddx_constants_type), intent(in) :: constants
45 type(ddx_workspace_type), intent(inout) :: workspace
46 type(ddx_state_type), intent(inout) :: state
47 type(ddx_error_type), intent(inout) :: error
48 real(dp), intent(in) :: phi_cav(constants % ncav), &
49 & psi(constants % nbasis, params % nsph), tol
50 real(dp), intent(out) :: esolv
51 real(dp), intent(out), optional :: force(3, params % nsph)
52
53 call ddpcm_setup(params, constants, workspace, state, phi_cav, psi, error)
54 if (error % flag .ne. 0) then
55 call update_error(error, &
56 & "ddlpb: ddpcm_setup returned an error, exiting")
57 return
58 end if
59 call ddpcm_guess(params, constants, workspace, state, error)
60 if (error % flag .ne. 0) then
61 call update_error(error, &
62 & "ddlpb: ddpcm_guess returned an error, exiting")
63 return
64 end if
65 call ddpcm_solve(params, constants, workspace, state, tol, error)
66 if (error % flag .ne. 0) then
67 call update_error(error, &
68 & "ddlpb: ddpcm_solve returned an error, exiting")
69 return
70 end if
71
72 call ddpcm_energy(constants, state, esolv, error)
73
74 ! Get forces if needed
75 if (params % force .eq. 1) then
76 ! solve the adjoint
77 call ddpcm_guess_adjoint(params, constants, workspace, state, error)
78 if (error % flag .ne. 0) then
79 call update_error(error, &
80 & "ddlpb: ddpcm_guess_adjoint returned an error, exiting")
81 return
82 end if
83 call ddpcm_solve_adjoint(params, constants, workspace, state, &
84 & tol, error)
85 if (error % flag .ne. 0) then
86 call update_error(error, &
87 & "ddlpb: ddpcm_guess_adjoint returned an error, exiting")
88 return
89 end if
90
91 ! evaluate the solvent unspecific contribution analytical derivatives
92 force = zero
93 call ddpcm_solvation_force_terms(params, constants, workspace, &
94 & state, force, error)
95 end if

252subroutine ddpcm_solve_adjoint(params, constants, workspace, state, tol, error)
253 implicit none
254 type(ddx_params_type), intent(in) :: params
255 type(ddx_constants_type), intent(in) :: constants
256 type(ddx_workspace_type), intent(inout) :: workspace
257 type(ddx_state_type), intent(inout) :: state
258 real(dp), intent(in) :: tol
259 type(ddx_error_type), intent(inout) :: error
260 ! local variables
261 real(dp) :: start_time, finish_time
262
263 state % s_niter = params % maxiter
264 start_time = omp_get_wtime()
265 call jacobi_diis(params, constants, workspace, tol, state % psi, &
266 & state % s, state % s_niter, state % s_rel_diff, lstarx, ldm1x, &
267 & hnorm, error)
268 finish_time = omp_get_wtime()
269 state % s_time = finish_time - start_time
270
271 if (error % flag .ne. 0) then
272 call update_error(error, "ddpcm_solve_adjoint: solver for ddCOSMO " // &
273 & "system did not converge, exiting")
274 return
275 end if
276
277 state % y_niter = params % maxiter
278 start_time = omp_get_wtime()
279 call jacobi_diis(params, constants, workspace, tol, state % s, state % y, &
280 & state % y_niter, state % y_rel_diff, repsstarx, prec_repsstarx, &
281 & hnorm, error)
282 finish_time = omp_get_wtime()
283 state % y_time = finish_time - start_time
284
285 if (error % flag .ne. 0) then
286 call update_error(error, "ddpcm_solve_adjoint: solver for ddPCM " // &
287 & "system did not converge, exiting")
288 return
289 end if
290
291 ! compute the real adjoint solution and store it in Q
292 state % q = state % s - fourpi/(params % eps - one)*state % y
293

308subroutine ddpcm_solvation_force_terms(params, constants, workspace, &
309 & state, force, error)
310 implicit none
311 type(ddx_params_type), intent(in) :: params
312 type(ddx_constants_type), intent(in) :: constants
313 type(ddx_workspace_type), intent(inout) :: workspace
314 type(ddx_state_type), intent(inout) :: state
315 type(ddx_error_type), intent(inout) :: error
316 real(dp), intent(out) :: force(3, params % nsph)
317
318 real(dp) :: start_time, finish_time
319 integer :: isph
320
321 start_time = omp_get_wtime()
322
323 call gradr(params, constants, workspace, state % g, state % ygrid, force)
324
325 do isph = 1, params % nsph
326 call contract_grad_l(params, constants, isph, state % xs, &
327 & state % sgrid, workspace % tmp_vylm(:, 1), &
328 & workspace % tmp_vdylm(:, :, 1), workspace % tmp_vplm(:, 1), &
329 & workspace % tmp_vcos(:, 1), workspace % tmp_vsin(:, 1), &
330 & force(:, isph))
331 call contract_grad_u(params, constants, isph, state % qgrid, &
332 & state % phi_grid, force(:, isph))
333 end do
334 force = - pt5 * force
335
336 finish_time = omp_get_wtime()
337 state % force_time = finish_time - start_time
338
339end subroutine ddpcm_solvation_force_terms

349subroutine ddpcm_derivative_setup(params, constants, workspace, state)
350 type(ddx_params_type), intent(in) :: params
351 type(ddx_constants_type), intent(in) :: constants
352 type(ddx_workspace_type), intent(inout) :: workspace
353 type(ddx_state_type), intent(inout) :: state
354
355 real(dp), external :: ddot
356 integer :: icav, isph, igrid
357
358 ! Get grid values of S and Y
359 call dgemm('T', 'N', params % ngrid, params % nsph, &
360 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
361 & state % s, constants % nbasis, zero, state % sgrid, params % ngrid)
362 call dgemm('T', 'N', params % ngrid, params % nsph, &
363 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
364 & state % y, constants % nbasis, zero, state % ygrid, params % ngrid)
365
366 ! Get the values of phi on the grid
367 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
368 & constants % icav_ia, constants % icav_ja, state % phi_cav, &
369 & state % phi_grid)
370
371 state % g = - (state % phieps - state % phi)
372 state % qgrid = state % sgrid - fourpi/(params % eps - one)*state % ygrid
373
374 ! assemble the intermediate zeta: S weighted by U evaluated on the
375 ! exposed grid points.
376 icav = 0
377 do isph = 1, params % nsph
378 do igrid = 1, params % ngrid
379 if (constants % ui(igrid, isph) .ne. zero) then
380 icav = icav + 1
381 state % zeta(icav) = constants % wgrid(igrid) * &
382 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
383 & constants % vgrid(1, igrid), 1, state % q(1, isph), 1)
384 end if
385 end do
386 end do
387
388end subroutine ddpcm_derivative_setup
