201  ! Check if solver did not converge

203  string = "ddpcm_energy: solver for ddCOSMO system did not converge"
204  call params % print_func(string)

206  end if

283 subroutine ddpcm_forces(params, constants, workspace, phi_grid, gradphi_cav, &
284  & psi, phi, phieps, s, sgrid, y, ygrid, g, q, qgrid, xs, zeta, force, info)
285  !! Inputs
286  type(ddx_params_type), intent(in) :: params
287  type(ddx_constants_type), intent(in) :: constants
288  real(dp), intent(in) :: phi_grid(params % ngrid, params % nsph), &
289  & gradphi_cav(3, constants % ncav), &
290  & psi(constants % nbasis, params % nsph), &
291  & phi(constants % nbasis, params % nsph), &
292  & phieps(constants % nbasis, params % nsph), &
293  & s(constants % nbasis, params % nsph), &
294  & y(constants % nbasis, params % nsph), &
295  & xs(constants % nbasis, params % nsph)
296  !! Temporary buffers
297  type(ddx_workspace_type), intent(inout) :: workspace
298  !! Outputs
299  real(dp), intent(out) :: sgrid(params % ngrid, params % nsph), &
300  & ygrid(params % ngrid, params % nsph), &
301  & g(constants % nbasis, params % nsph), &
302  & q(constants % nbasis, params % nsph), &
303  & qgrid(params % ngrid, params % nsph), &
304  & zeta(constants % ncav), &
305  & force(3, params % nsph)
306  integer, intent(out) :: info
307  !! Local variables
308  integer :: isph, icav, igrid, inode, jnode, jsph, jnear
309  real(dp) :: tmp1, tmp2, d(3), dnorm
310  real(dp), external :: ddot, dnrm2
311  !! The code
312  ! Get grid values of S and Y
313  call dgemm('T', 'N', params % ngrid, params % nsph, &
314  & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
315  & s, constants % nbasis, zero, sgrid, params % ngrid)
316  call dgemm('T', 'N', params % ngrid, params % nsph, &
317  & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
318  & y, constants % nbasis, zero, ygrid, params % ngrid)
319  g = phieps - phi
320  q = s - fourpi/(params % eps-one)*y
321  qgrid = sgrid - fourpi/(params % eps-one)*ygrid
322  ! gradr initializes forces with zeros
323  call gradr(params, constants, workspace, g, ygrid, force)
324  do isph = 1, params % nsph
325  call fdoka(params, constants, isph, xs, sgrid(:, isph), &
326  & workspace % tmp_vylm(:, 1), workspace % tmp_vdylm(:, :, 1), &
327  & workspace % tmp_vplm(:, 1), workspace % tmp_vcos(:, 1), &
328  & workspace % tmp_vsin(:, 1), force(:, isph))
329  call fdokb(params, constants, isph, xs, sgrid, &
330  & workspace % tmp_vylm(:, 1), workspace % tmp_vdylm(:, :, 1), &
331  & workspace % tmp_vplm(:, 1), workspace % tmp_vcos(:, 1), &
332  & workspace % tmp_vsin(:, 1), force(:, isph))
333  call fdoga(params, constants, isph, qgrid, phi_grid, force(:, isph))
334  end do
335  force = -pt5 * force
336  icav = 0
337  do isph = 1, params % nsph
338  do igrid = 1, params % ngrid
339  if(constants % ui(igrid, isph) .ne. zero) then
340  icav = icav + 1
341  zeta(icav) = -pt5 * constants % wgrid(igrid) * &
342  & constants % ui(igrid, isph) * ddot(constants % nbasis, &
343  & constants % vgrid(1, igrid), 1, q(1, isph), 1)
344  force(:, isph) = force(:, isph) + &
345  & zeta(icav)*gradphi_cav(:, icav)
346  end if
347  end do
348  end do
349  !! Last term where we compute gradients of potential at centers of atoms
350  !! spawned by intermediate zeta.
351  if(params % fmm .eq. 1) then
352  !! This step can be substituted by a proper dgemm if zeta
353  !! intermediate is converted from cavity points to all grid points
354  !! with zero values at internal grid points
355  ! P2M step
356  icav = 0
357  do isph = 1, params % nsph
358  inode = constants % snode(isph)
359  workspace % tmp_node_m(:, inode) = zero
360  do igrid = 1, params % ngrid
361  if(constants % ui(igrid, isph) .eq. zero) cycle
362  icav = icav + 1
363  call fmm_p2m(constants % cgrid(:, igrid), zeta(icav), &
364  & one, params % pm, constants % vscales, one, &
365  & workspace % tmp_node_m(:, inode))
366  end do
367  workspace % tmp_node_m(:, inode) = workspace % tmp_node_m(:, inode) / &
368  & params % rsph(isph)
369  end do
370  ! M2M, M2L and L2L translations
371  call tree_m2m_rotation(params, constants, workspace % tmp_node_m)
372  call tree_m2l_rotation(params, constants, workspace % tmp_node_m, &
373  & workspace % tmp_node_l)
374  call tree_l2l_rotation(params, constants, workspace % tmp_node_l)
375  ! Now compute near-field FMM gradients
376  ! Cycle over all spheres
377  icav = 0
378  workspace % tmp_efld = zero
379  do isph = 1, params % nsph
380  ! Cycle over all external grid points
381  do igrid = 1, params % ngrid
382  if(constants % ui(igrid, isph) .eq. zero) cycle
383  icav = icav + 1
384  ! Cycle over all near-field admissible pairs of spheres,
385  ! including pair (isph, isph) which is a self-interaction
386  inode = constants % snode(isph)
387  do jnear = constants % snear(inode), constants % snear(inode+1)-1
388  ! Near-field interactions are possible only between leaf
389  ! nodes, which must contain only a single input sphere
390  jnode = constants % near(jnear)
391  jsph = constants % order(constants % cluster(1, jnode))
392  d = params % csph(:, isph) + &
393  & constants % cgrid(:, igrid)*params % rsph(isph) - &
394  & params % csph(:, jsph)
395  dnorm = dnrm2(3, d, 1)
396  workspace % tmp_efld(:, jsph) = workspace % tmp_efld(:, jsph) + &
397  & zeta(icav)*d/(dnorm**3)
398  end do
399  end do
400  end do
401  ! Take into account far-field FMM gradients only if pl > 0
402  if (params % pl .gt. 0) then
403  tmp1 = one / sqrt(3d0) / constants % vscales(1)
404  do isph = 1, params % nsph
405  inode = constants % snode(isph)
406  tmp2 = tmp1 / params % rsph(isph)
407  workspace % tmp_efld(3, isph) = workspace % tmp_efld(3, isph) + &
408  & tmp2*workspace % tmp_node_l(3, inode)
409  workspace % tmp_efld(1, isph) = workspace % tmp_efld(1, isph) + &
410  & tmp2*workspace % tmp_node_l(4, inode)
411  workspace % tmp_efld(2, isph) = workspace % tmp_efld(2, isph) + &
412  & tmp2*workspace % tmp_node_l(2, inode)
413  end do
414  end if
415  do isph = 1, params % nsph
416  force(:, isph) = force(:, isph) + &
417  & workspace % tmp_efld(:, isph)*params % charge(isph)
418  end do
419  ! Naive quadratically scaling implementation
420  else
421  ! This routines actually computes -grad, not grad
422  call efld(constants % ncav, zeta, constants % ccav, params % nsph, &
423  & params % csph, workspace % tmp_efld)
424  do isph = 1, params % nsph
425  force(:, isph) = force(:, isph) - &
426  & workspace % tmp_efld(:, isph)*params % charge(isph)
427  end do
428  end if
429  ! Clear status
430  info = 0
431 end subroutine ddpcm_forces
