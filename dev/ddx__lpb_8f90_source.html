64 subroutine ddlpb(ddx_data, phi_cav, gradphi_cav, hessianphi_cav, psi, tol, esolv, &
65  & force, info)
66  ! main ddLPB
67  ! Inputs
68  type(ddx_type), intent(inout) :: ddx_data
69  real(dp), dimension(ddx_data % constants % ncav), intent(in) :: phi_cav
70  real(dp), dimension(3, ddx_data % constants % ncav), intent(in) :: gradphi_cav
71  real(dp), dimension(3,3, ddx_data % constants % ncav), intent(in) :: hessianphi_cav
72  real(dp), dimension(ddx_data % constants % nbasis, & & ddx_data % params % nsph), intent(in) :: psi
73  real(dp), intent(in) :: tol
74  ! Outputs
75  real(dp), intent(out) :: esolv
76  real(dp), dimension(3, ddx_data % params % nsph), intent(out) :: force
77  ! internal
78  integer, intent(out) :: info
79  integer :: isph, igrid, istatus
80  !
81  ! x(:,:,1): X_r Reaction potential solution (Laplace equation)
82  ! x(:,:,2): X_e Extended potential solution (HSP equation)
83  !
84 
85  !! Setting initial values to zero
86  ddx_data % g_lpb = zero
87  ddx_data % f_lpb = zero
88  ddx_data % phi_grid = zero
89 
90  ! Unwrap sparsely stored potential at cavity points phi_cav into phi_grid
91  ! and multiply it by characteristic function at cavity points ui
92  call ddcav_to_grid_work(ddx_data % params % ngrid, ddx_data % params % nsph, &
93  & ddx_data % constants % ncav, ddx_data % constants % icav_ia, &
94  & ddx_data % constants % icav_ja, phi_cav, ddx_data % phi_grid)
95  ddx_data % workspace % tmp_cav = phi_cav * ddx_data % constants % ui_cav
96  call ddcav_to_grid_work(ddx_data % params % ngrid, ddx_data % params % nsph, &
97  & ddx_data % constants % ncav, ddx_data % constants % icav_ia, &
98  & ddx_data % constants % icav_ja, ddx_data % workspace % tmp_cav, &
99  & ddx_data % workspace % tmp_grid)
100  ddx_data % g_lpb = - ddx_data % workspace % tmp_grid
101 
102  ! wghpot_f : Intermediate computation of F_0 Eq.(75) from QSM19.SISC
103  call wghpot_f(ddx_data % params, ddx_data % constants, &
104  & ddx_data % workspace, gradphi_cav, ddx_data % f_lpb)
105 
106  ! use a tighter tolerance for microiterations
107  ddx_data % constants % inner_tol = tol/100.0d0
108 
109  ddx_data % xs_niter = ddx_data % params % maxiter
110  ! Call the subroutine to solve for Esolv
111  call ddlpb_energy(ddx_data % params, ddx_data % constants, &
112  & ddx_data % workspace, ddx_data % g_lpb, ddx_data % f_lpb, &
113  & ddx_data % x_lpb, ddx_data % xs_niter, ddx_data % xs_rel_diff, ddx_data % xs_time, &
114  & tol, esolv, info)
115 
116  ! Start the Force computation
117  if(ddx_data % params % force .eq. 1) then
118  ddx_data % s_niter = ddx_data % params % maxiter
119  ! Call the subroutine adjoint to solve the adjoint solution
120  call ddlpb_adjoint(ddx_data % params, ddx_data % constants, &
121  & ddx_data % workspace, psi, tol, ddx_data % x_adj_lpb, &
122  ddx_data % s_niter, ddx_data % s_rel_diff, ddx_data % s_time, info)
123 
124  !Call the subroutine to evaluate derivatives
125  call ddlpb_force(ddx_data % params, ddx_data % constants, &
126  & ddx_data % workspace, hessianphi_cav, ddx_data % phi_grid, gradphi_cav, &
127  & ddx_data % x_lpb, ddx_data % x_adj_lpb, ddx_data % zeta, force, info)
128 
129  endif
130 
131 ! deallocate(x, x_adj, g, f, phi_grid, stat = istatus)
132 ! if (istatus.ne.0) write(6,*) 'ddlpb deallocation failed'
133 
134 end subroutine ddlpb

150 subroutine ddlpb_energy(params, constants, workspace, g, f, &
151  & x, n_iter, xs_rel_diff, xs_time, tol , esolv, info)
152  type(ddx_params_type), intent(in) :: params
153  type(ddx_constants_type), intent(in) :: constants
154  type(ddx_workspace_type), intent(inout) :: workspace
155  real(dp), dimension(params % ngrid, params % nsph), intent(in) :: g, f
156  real(dp), dimension(constants % nbasis, params % nsph, 2), intent(out) :: x
157  real(dp), intent(in) :: tol
158  integer, intent(inout) :: n_iter
159  !! Outputs
160  real(dp), intent(out) :: xs_rel_diff(n_iter), xs_time
161  real(dp), intent(out) :: esolv
162  integer, intent(out) :: info
163  integer :: isph, istat
164  real(dp) :: start_time, finish_time
165  real(dp), allocatable :: rhs(:,:,:)
166 
167  allocate(rhs(constants % nbasis, params % nsph, 2), stat=istat)
168  if (istat.ne.0) stop 1
169 
170  ! Setting of the local variables
171  rhs = zero
172 
173  ! integrate RHS
174  call ddintegrate(params % nsph, constants % nbasis, &
175  & params % ngrid, constants % vwgrid, &
176  & constants % vgrid_nbasis, g, rhs(:,:,1))
177  call ddintegrate(params % nsph, constants % nbasis, &
178  & params % ngrid, constants % vwgrid, &
179  & constants % vgrid_nbasis, f, rhs(:,:,2))
180  rhs(:,:,1) = rhs(:,:,1) + rhs(:,:,2)
181 
182  ! guess
183  workspace % ddcosmo_guess = zero
184  workspace % hsp_guess = zero
185  call lpb_direct_prec(params, constants, workspace, rhs, x)
186 
187  ! solve LS using Jacobi/DIIS
188  call cpu_time(start_time)
189  call jacobi_diis_external(params, constants, workspace, 2*constants % n, &
190  & tol, rhs, x, n_iter, xs_rel_diff, lpb_direct_matvec, &
191  & lpb_direct_prec, rmsnorm, info)
192  call cpu_time(finish_time)
193  xs_time = finish_time - start_time
194 
195  esolv = zero
196  do isph = 1, params % nsph
197  esolv = esolv + pt5*params % charge(isph)*x(1,isph,1)*(one/sqrt4pi)
198  end do
199  deallocate(rhs)
200  if (istat.ne.0) stop 1
201 end subroutine ddlpb_energy

213 subroutine ddlpb_adjoint(params, constants, workspace, psi, tol, x_adj, &
214  & n_iter, s_rel_diff, s_time, info)
215  implicit none
216  type(ddx_params_type), intent(in) :: params
217  type(ddx_constants_type), intent(in) :: constants
218  real(dp), dimension(constants % nbasis, params % nsph), intent(in) :: psi
219  real(dp), intent(in) :: tol
220  !!Output
221  integer, intent(inout) :: n_iter
222  type(ddx_workspace_type), intent(inout) :: workspace
223  real(dp), dimension(constants % nbasis, params % nsph, 2), intent(out) :: x_adj
224  real(dp), intent(out) :: s_time
225  real(dp), intent(out) :: s_rel_diff(n_iter)
226  integer, intent(out) :: info
227 
228  real(dp), allocatable :: rhs(:,:,:)
229  integer :: istat
230  real(dp) :: start_time, finish_time
231 
232  allocate(rhs(constants % nbasis, params % nsph, 2), stat=istat)
233  if (istat.ne.0) stop 1
234 
235  ! set up the RHS
236  rhs(:,:,1) = psi
237  rhs(:,:,2) = zero
238 
239  ! guess
240  workspace % ddcosmo_guess = zero
241  workspace % hsp_guess = zero

243 
244  call cpu_time(start_time)

251

253  if (istat.ne.0) stop 1

270 subroutine ddlpb_force(params, constants, workspace, hessian, phi_grid, gradphi, &
271  & x, x_adj, zeta, force, info)
272  !! input/output
273  type(ddx_params_type), intent(in) :: params
274  type(ddx_constants_type), intent(in) :: constants
275  type(ddx_workspace_type), intent(inout) :: workspace
276 
277  real(dp), dimension(3, 3, constants % ncav), intent(in) :: hessian
278  real(dp), dimension(params % ngrid, params % nsph), intent(in) :: phi_grid
279  real(dp), dimension(3, constants % ncav), intent(in) :: gradphi
280  real(dp), dimension(constants % nbasis, params % nsph, 2), intent(in) :: x, x_adj
281  real(dp), dimension(3, params % nsph), intent(out) :: force
282  real(dp), intent(out) :: zeta(constants % ncav)
283  integer, intent(out) :: info
284 
285  ! local
286  real(dp), dimension(constants % nbasis) :: basloc, vplm
287  real(dp), dimension(3, constants % nbasis) :: dbasloc
288  real(dp), dimension(params % lmax + 1) :: vsin, vcos
289 
290  ! large local are allocatable
291  real(dp), allocatable :: ef(:,:), xadj_r_sgrid(:,:), xadj_e_sgrid(:,:), &
292  & normal_hessian_cav(:,:), diff_re(:,:), scaled_xr(:,:)
293  integer :: isph, icav, icav_gr, icav_ge, igrid, istat
294  integer :: i, inode, jnear, inear, jnode, jsph
295  real(dp), external :: ddot, dnrm2
296  real(dp) :: tcontract_gradi_Lik, tcontract_gradi_Lji, tcontract_gradi_Bik, tcontract_gradi_Bji, tcontract_grad_U, &
297  & tcontract_grad_C_worker1, tcontract_grad_C_worker2
298  real(dp) :: d(3), dnorm, tmp1, tmp2
299 
300  allocate(ef(3, params % nsph), &
301  & xadj_r_sgrid(params % ngrid, params % nsph), &
302  & xadj_e_sgrid(params % ngrid, params % nsph), &
303  & normal_hessian_cav(3, constants % ncav), &
304  & diff_re(constants % nbasis, params % nsph), &
305  & scaled_xr(constants % nbasis, params % nsph), stat=istat)
306  if (istat.ne.0) stop 1
307 
308  diff_re = zero
309  vsin = zero
310  vcos = zero
311  vplm = zero
312  basloc = zero
313  dbasloc = zero
314  ef = zero
315  force = zero
316 
317  ! Compute the derivative of the normal derivative of psi_0
318  normal_hessian_cav = zero
319  icav = 0
320  do isph = 1, params % nsph
321  do igrid = 1, params % ngrid
322  if(constants % ui(igrid, isph) .gt. zero) then
323  icav = icav + 1
324  do i = 1, 3
325  normal_hessian_cav(:, icav) = normal_hessian_cav(:,icav) +&
326  & hessian(:,i,icav)*constants % cgrid(i,igrid)
327  end do
328  end if
329  end do
330  end do
331 
332  ! Call dgemm to integrate the adjoint solution on the grid points
333  call dgemm('T', 'N', params % ngrid, params % nsph, constants % nbasis, &
334  & one, constants % vgrid, constants % vgrid_nbasis, x_adj(:,:,1), &
335  & constants % nbasis, zero, xadj_r_sgrid, params % ngrid)
336  call dgemm('T', 'N', params % ngrid, params % nsph, constants % nbasis, &
337  & one, constants % vgrid, constants % vgrid_nbasis, x_adj(:,:,2), &
338  & constants % nbasis, zero, xadj_e_sgrid, params % ngrid)
339 
340  ! Scale by the factor of 1/(4Pi/(2l+1))
341  scaled_xr = x(:,:,1)
342  call convert_ddcosmo(params, constants, -1, scaled_xr)
343 
344  tcontract_gradi_lik = zero
345  tcontract_gradi_lji = zero
346  tcontract_gradi_bik = zero
347  tcontract_gradi_bji = zero
348  tcontract_grad_c_worker2 = zero
349  tcontract_grad_c_worker1 = zero
350  tcontract_grad_u = zero
351  do isph = 1, params % nsph
352  ! Compute A^k*Xadj_r, using Subroutine from ddCOSMO
353  call contract_grad_l(params, constants, isph, scaled_xr, xadj_r_sgrid, &
354  & basloc, dbasloc, vplm, vcos, vsin, force(:,isph))
355  ! Compute B^k*Xadj_e
356  call contract_grad_b(params, constants, workspace, isph, x(:,:,2), xadj_e_sgrid, &
357  & basloc, dbasloc, vplm, vcos, vsin, force(:, isph))
358  ! Computation of G0
359  call contract_grad_u(params, constants, isph, xadj_r_sgrid, phi_grid, &
360  & force(:, isph))
361  end do
362  ! Compute C1 and C2 contributions
363  diff_re = zero
364  call contract_grad_c(params, constants, workspace, x(:,:,1), x(:,:,2), xadj_r_sgrid, &
365  & xadj_e_sgrid, x_adj(:,:,1), x_adj(:,:,2), force, diff_re)
366  ! Computation of G0 continued
367  ! NOTE: contract_grad_U returns a positive summation
368  force = -force
369  icav = 0
370  do isph = 1, params % nsph
371  do igrid = 1, params % ngrid
372  if(constants % ui(igrid, isph) .ne. zero) then
373  icav = icav + 1
374  zeta(icav) = -constants % wgrid(igrid) * &
375  & constants % ui(igrid, isph) * ddot(constants % nbasis, &
376  & constants % vgrid(1, igrid), 1, &
377  & x_adj(1, isph, 1), 1)
378  force(:, isph) = force(:, isph) + &
379  & zeta(icav)*gradphi(:, icav)
380  end if
381  end do
382  end do
383  !! Last term where we compute gradients of potential at centers of atoms
384  !! spawned by intermediate zeta.
385  if(params % fmm .eq. 1) then
386  !! This step can be substituted by a proper dgemm if zeta
387  !! intermediate is converted from cavity points to all grid points
388  !! with zero values at internal grid points
389  ! P2M step
390  icav = 0
391  do isph = 1, params % nsph
392  inode = constants % snode(isph)
393  workspace % tmp_node_m(:, inode) = zero
394  do igrid = 1, params % ngrid
395  if(constants % ui(igrid, isph) .eq. zero) cycle
396  icav = icav + 1
397  call fmm_p2m(constants % cgrid(:, igrid), zeta(icav), &
398  & one, params % pm, constants % vscales, one, &
399  & workspace % tmp_node_m(:, inode))
400  end do
401  workspace % tmp_node_m(:, inode) = workspace % tmp_node_m(:, inode) / &
402  & params % rsph(isph)
403  end do
404  ! M2M, M2L and L2L translations
405  call tree_m2m_rotation(params, constants, workspace % tmp_node_m)
406  call tree_m2l_rotation(params, constants, workspace % tmp_node_m, &
407  & workspace % tmp_node_l)
408  call tree_l2l_rotation(params, constants, workspace % tmp_node_l)
409  ! Now compute near-field FMM gradients
410  ! Cycle over all spheres
411  icav = 0
412  workspace % tmp_efld = zero
413  do isph = 1, params % nsph
414  ! Cycle over all external grid points
415  do igrid = 1, params % ngrid
416  if(constants % ui(igrid, isph) .eq. zero) cycle
417  icav = icav + 1
418  ! Cycle over all near-field admissible pairs of spheres,
419  ! including pair (isph, isph) which is a self-interaction
420  inode = constants % snode(isph)
421  do jnear = constants % snear(inode), constants % snear(inode+1)-1
422  ! Near-field interactions are possible only between leaf
423  ! nodes, which must contain only a single input sphere
424  jnode = constants % near(jnear)
425  jsph = constants % order(constants % cluster(1, jnode))
426  d = params % csph(:, isph) + &
427  & constants % cgrid(:, igrid)*params % rsph(isph) - &
428  & params % csph(:, jsph)
429  dnorm = dnrm2(3, d, 1)
430  workspace % tmp_efld(:, jsph) = workspace % tmp_efld(:, jsph) + &
431  & zeta(icav)*d/(dnorm**3)
432  end do
433  end do
434  end do
435  ! Take into account far-field FMM gradients only if pl > 0
436  if (params % pl .gt. 0) then
437  tmp1 = one / sqrt(3d0) / constants % vscales(1)
438  do isph = 1, params % nsph
439  inode = constants % snode(isph)
440  tmp2 = tmp1 / params % rsph(isph)
441  workspace % tmp_efld(3, isph) = workspace % tmp_efld(3, isph) + &
442  & tmp2*workspace % tmp_node_l(3, inode)
443  workspace % tmp_efld(1, isph) = workspace % tmp_efld(1, isph) + &
444  & tmp2*workspace % tmp_node_l(4, inode)
445  workspace % tmp_efld(2, isph) = workspace % tmp_efld(2, isph) + &
446  & tmp2*workspace % tmp_node_l(2, inode)
447  end do
448  end if
449  do isph = 1, params % nsph
450  force(:, isph) = force(:, isph) + &
451  & workspace % tmp_efld(:, isph)*params % charge(isph)
452  end do
453  ! Naive quadratically scaling implementation
454  else
455  ! This routines actually computes -grad, not grad
456  call efld(constants % ncav, zeta, constants % ccav, params % nsph, &
457  & params % csph, workspace % tmp_efld)
458  do isph = 1, params % nsph
459  force(:, isph) = force(:, isph) - &
460  & workspace % tmp_efld(:, isph)*params % charge(isph)
461  end do
462  end if
463 
464  tcontract_grad_c_worker2 = zero
465  tcontract_grad_u = zero
466  icav_gr = zero
467  icav_ge = zero
468  ! Computation of F0
469  call contract_grad_f(params, constants, workspace, x_adj(:,:,1), xadj_r_sgrid, &
470  & gradphi, normal_hessian_cav, icav_gr, force)
471  call contract_grad_f(params, constants, workspace, x_adj(:,:,2), xadj_e_sgrid, &
472  & gradphi, normal_hessian_cav, icav_ge, force)
473 
474  force = pt5*force
475 
476  deallocate(ef, xadj_r_sgrid, xadj_e_sgrid, normal_hessian_cav, &
477  & diff_re, scaled_xr, stat=istat)
478  if (istat.ne.0) stop 1
479 end subroutine ddlpb_force
