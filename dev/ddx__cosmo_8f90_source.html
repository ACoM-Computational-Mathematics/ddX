137 subroutine ddcosmo_solve_worker(params, constants, workspace, phi_cav, &
138  & xs, xs_niter, xs_rel_diff, xs_time, tol, phi_grid, &
139  & phi, info)
140  !! Inputs
141  type(ddx_params_type), intent(in) :: params
142  type(ddx_constants_type), intent(in) :: constants
143  real(dp), intent(in) :: phi_cav(constants % ncav), tol
144  !! Input+output
145  real(dp), intent(inout) :: xs(constants % nbasis, params % nsph)
146  integer, intent(inout) :: xs_niter
147  !! Temporary buffers
148  type(ddx_workspace_type), intent(inout) :: workspace
149  !! Outputs
150  real(dp), intent(out) :: xs_rel_diff(xs_niter), xs_time
151  real(dp), intent(out) :: phi_grid(params % ngrid, params % nsph)
152  real(dp), intent(out) :: phi(constants % nbasis, params % nsph)
153  integer, intent(out) :: info
154  !! Local variables
155  character(len=255) :: string
156  real(dp) :: start_time, finish_time, r_norm
157  !! The code
158  ! At first check if parameters, constants and workspace are correctly
159  ! initialized
160  if (params % error_flag .ne. 0) then
161  string = "ddcosmo_energy: `params` is in error state"
162  call params % print_func(string)
163  info = 1
164  return
165  end if
166  if (constants % error_flag .ne. 0) then
167  string = "ddcosmo_energy: `constants` is in error state"
168  call params % print_func(string)
169  info = 1
170  return
171  end if
172  if (workspace % error_flag .ne. 0) then
173  string = "ddcosmo_energy: `workspace` is in error state"
174  call params % print_func(string)
175  info = 1
176  return
177  end if
178  ! Unwrap sparsely stored potential at cavity points phi_cav into phi_grid
179  ! and multiply it by characteristic function at cavity points ui
180  call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
181  & constants % icav_ia, constants % icav_ja, phi_cav, phi_grid)
182  workspace % tmp_cav = phi_cav * constants % ui_cav
183  call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
184  & constants % icav_ia, constants % icav_ja, workspace % tmp_cav, &
185  & workspace % tmp_grid)
186  ! Integrate against spherical harmonics and Lebedev weights to get Phi
187  call ddintegrate(params % nsph, constants % nbasis, &
188  & params % ngrid, constants % vwgrid, &
189  & constants % vgrid_nbasis, workspace % tmp_grid, phi)
190 ! call ddintegrate_sph_work(constants % nbasis, params % ngrid, &
191 ! & params % nsph, constants % vwgrid, constants % vgrid_nbasis, &
192 ! & one, workspace % tmp_grid, zero, phi)
193  ! Set right hand side to -Phi
194  workspace % tmp_rhs = -phi
195  ! Solve ddCOSMO system L X = -Phi with a given initial guess
196  start_time = omp_get_wtime()
197  if (params % itersolver .eq. 1) then
198  call jacobi_diis(params, constants, workspace, tol, &
199  & workspace % tmp_rhs, xs, xs_niter, xs_rel_diff, lx, &
200  & ldm1x, hnorm, info)
201  else
202  call gmresr(params, constants, workspace, tol, &
203  & workspace % tmp_rhs, xs, xs_niter, r_norm, lx, info)
204  end if
205  finish_time = omp_get_wtime()
206  xs_time = finish_time - start_time
207  ! Check if solver did not converge
208  if (info .ne. 0) then
209  string = "ddcosmo_energy: solver for ddCOSMO system did not converge"
210  call params % print_func(string)
211  return
212  end if
213  ! Clear status
214  info = 0
215 end subroutine ddcosmo_solve_worker

285 subroutine ddcosmo_forces_worker(params, constants, workspace, phi_grid, &
286  & gradphi_cav, psi, s, sgrid, xs, zeta, force, info)
287  !! Inputs
288  type(ddx_params_type), intent(in) :: params
289  type(ddx_constants_type), intent(in) :: constants
290  real(dp), intent(in) :: phi_grid(params % ngrid, params % nsph), &
291  & gradphi_cav(3, constants % ncav), &
292  & psi(constants % nbasis, params % nsph), &
293  & s(constants % nbasis, params % nsph), &
294  & xs(constants % nbasis, params % nsph)
295  !! Temporary buffers
296  type(ddx_workspace_type), intent(inout) :: workspace
297  !! Outputs
298  real(dp), intent(out) :: sgrid(params % ngrid, params % nsph), &
299  & zeta(constants % ncav), &
300  & force(3, params % nsph)
301  integer, intent(out) :: info
302  !! Local variables
303  integer :: isph, icav, igrid, inode, jnode, jsph, jnear
304  real(dp) :: tmp1, tmp2, d(3), dnorm
305  real(dp), external :: ddot, dnrm2
306  character(len=255) :: string
307  !! The code
308  ! At first check if parameters, constants and workspace are correctly
309  ! initialized
310  if (params % error_flag .ne. 0) then
311  string = "ddcosmo_forces: `params` is in error state"
312  call params % print_func(string)
313  info = 1
314  return
315  end if
316  if (constants % error_flag .ne. 0) then
317  string = "ddcosmo_forces: `constants` is in error state"
318  call params % print_func(string)
319  info = 1
320  return
321  end if
322  if (workspace % error_flag .ne. 0) then
323  string = "ddcosmo_forces: `workspace` is in error state"
324  call params % print_func(string)
325  info = 1
326  return
327  end if
328  ! Get values of S on grid
329  call ddeval_grid_work(constants % nbasis, params % ngrid, params % nsph, &
330  & constants % vgrid, constants % vgrid_nbasis, one, s, zero, sgrid)
331  force = zero
332  do isph = 1, params % nsph
333  call contract_grad_l(params, constants, isph, xs, sgrid, &
334  & workspace % tmp_vylm(:, 1), workspace % tmp_vdylm(:, :, 1), &
335  & workspace % tmp_vplm(:, 1), &
336  & workspace % tmp_vcos(:, 1), &
337  & workspace % tmp_vsin(:, 1), force(:, isph))
338  call contract_grad_u(params, constants, isph, sgrid, phi_grid, &
339  & force(:, isph))
340  end do
341  force = -pt5 * force
342  icav = 0
343  do isph = 1, params % nsph
344  do igrid = 1, params % ngrid
345  if (constants % ui(igrid, isph) .ne. zero) then
346  icav = icav + 1
347  zeta(icav) = -pt5 * constants % wgrid(igrid) * &
348  & constants % ui(igrid, isph) * ddot(constants % nbasis, &
349  & constants % vgrid(1, igrid), 1, &
350  & s(1, isph), 1)
351  force(:, isph) = force(:, isph) + &
352  & zeta(icav)*gradphi_cav(:, icav)
353  end if
354  end do
355  end do
356  !! Last term where we compute gradients of potential at centers of atoms
357  !! spawned by intermediate zeta.
358  if(params % fmm .eq. 1) then
359  !! This step can be substituted by a proper dgemm if zeta
360  !! intermediate is converted from cavity points to all grid points
361  !! with zero values at internal grid points
362  ! P2M step
363  icav = 0
364  do isph = 1, params % nsph
365  inode = constants % snode(isph)
366  workspace % tmp_node_m(:, inode) = zero
367  do igrid = 1, params % ngrid
368  if(constants % ui(igrid, isph) .eq. zero) cycle
369  icav = icav + 1
370  call fmm_p2m(constants % cgrid(:, igrid), zeta(icav), &
371  & one, params % pm, constants % vscales, one, &
372  & workspace % tmp_node_m(:, inode))
373  end do
374  workspace % tmp_node_m(:, inode) = workspace % tmp_node_m(:, inode) / &
375  & params % rsph(isph)
376  end do
377  ! M2M, M2L and L2L translations
378  call tree_m2m_rotation(params, constants, workspace % tmp_node_m)
379  call tree_m2l_rotation(params, constants, workspace % tmp_node_m, &
380  & workspace % tmp_node_l)
381  call tree_l2l_rotation(params, constants, workspace % tmp_node_l)
382  ! Now compute near-field FMM gradients
383  ! Cycle over all spheres
384  icav = 0
385  workspace % tmp_efld = zero
386  do isph = 1, params % nsph
387  ! Cycle over all external grid points
388  do igrid = 1, params % ngrid
389  if (constants % ui(igrid, isph) .eq. zero) cycle
390  icav = icav + 1
391  ! Cycle over all near-field admissible pairs of spheres,
392  ! including pair (isph, isph) which is a self-interaction
393  inode = constants % snode(isph)
394  do jnear = constants % snear(inode), constants % snear(inode+1)-1
395  ! Near-field interactions are possible only between leaf
396  ! nodes, which must contain only a single input sphere
397  jnode = constants % near(jnear)
398  jsph = constants % order(constants % cluster(1, jnode))
399  d = params % csph(:, isph) + &
400  & constants % cgrid(:, igrid)*params % rsph(isph) - &
401  & params % csph(:, jsph)
402  dnorm = dnrm2(3, d, 1)
403  workspace % tmp_efld(:, jsph) = workspace % tmp_efld(:, jsph) + &
404  & zeta(icav)*d/(dnorm**3)
405  end do
406  end do
407  end do
408  ! Take into account far-field FMM gradients only if pl > 0
409  if (params % pl .gt. 0) then
410  tmp1 = one / sqrt(3d0) / constants % vscales(1)
411  do isph = 1, params % nsph
412  inode = constants % snode(isph)
413  tmp2 = tmp1 / params % rsph(isph)
414  workspace % tmp_efld(3, isph) = workspace % tmp_efld(3, isph) + &
415  & tmp2*workspace % tmp_node_l(3, inode)
416  workspace % tmp_efld(1, isph) = workspace % tmp_efld(1, isph) + &
417  & tmp2*workspace % tmp_node_l(4, inode)
418  workspace % tmp_efld(2, isph) = workspace % tmp_efld(2, isph) + &
419  & tmp2*workspace % tmp_node_l(2, inode)
420  end do
421  end if
422  do isph = 1, params % nsph
423  force(:, isph) = force(:, isph) + &
424  & workspace % tmp_efld(:, isph)*params % charge(isph)
425  end do
426  ! Naive quadratically scaling implementation
427  else
428  ! This routines actually computes -grad, not grad
429  call efld(constants % ncav, zeta, constants % ccav, &
430  & params % nsph, params % csph, workspace % tmp_efld)
431  do isph = 1, params % nsph
432  force(:, isph) = force(:, isph) - &
433  & workspace % tmp_efld(:, isph)*params % charge(isph)
434  end do
435  end if
436  ! Clear status
437  info = 0
438 end subroutine ddcosmo_forces_worker
