1081 real(dp) function fsw(t, se, eta)
1082  ! Inputs
1083  real(dp), intent(in) :: t, se, eta
1084  ! Local variables
1085  real(dp) :: a, b, c, x
1086  ! Apply shift:
1087  ! se = 0 => t - eta/2 [ CENTERED ]
1088  ! se = 1 => t - eta [ EXTERIOR ]
1089  ! se = -1 => t [ INTERIOR ]
1090  x = t - (se*pt5 + pt5)*eta
1091  ! a must be in range (0,eta) for the switch region
1092  a = one - x
1093  ! Define switch function chi
1094  ! a <= 0 corresponds to the outside region
1095  if (a .le. zero) then
1096  fsw = zero
1097  ! a >= eta corresponds to the inside region
1098  else if (a .ge. eta) then
1099  fsw = one
1100  ! Switch region wher x is in range (1-eta,1)
1101  else
1102  ! Normalize a to region (0,1)
1103  a = a / eta
1104  b = 6d0*a - 15
1105  c = b*a + 10
1106  fsw = a * a * a * c
1107  end if
1108 end function fsw

1136 real(dp) function dfsw(t, se, eta)
1137  ! Inputs
1138  real(dp), intent(in) :: t, se, eta
1139  ! Local variables
1140  real(dp) :: flow, x
1141  real(dp), parameter :: f30=30.0d0
1142  ! Apply shift:
1143  ! s = 0 => t - eta/2 [ CENTERED ]
1144  ! s = 1 => t - eta [ EXTERIOR ]
1145  ! s = -1 => t [ INTERIOR ]
1146  x = t - (se + 1.d0)*eta / 2.d0
1147  ! Lower bound of switch region
1148  flow = one - eta
1149  ! Define derivative of switch function chi
1150  if (x .ge. one) then
1151  dfsw = zero
1152  else if (x .le. flow) then
1153  dfsw = zero
1154  else
1155  dfsw = -f30 * (( (one-x)*(x-one+eta) )**2) / (eta**5)
1156  endif
1157 end function dfsw

1259 subroutine tree_rib_build(nsph, csph, rsph, order, cluster, children, parent, &
1260  & cnode, rnode, snode)
1261  ! Inputs
1262  integer, intent(in) :: nsph
1263  real(dp), intent(in) :: csph(3, nsph), rsph(nsph)
1264  ! Outputs
1265  integer, intent(out) :: order(nsph), cluster(2, 2*nsph-1), &
1266  & children(2, 2*nsph-1), parent(2*nsph-1), snode(nsph)
1267  real(dp), intent(out) :: cnode(3, 2*nsph-1), rnode(2*nsph-1)
1268  ! Local variables
1269  integer :: nclusters, i, j, n, s, e, div
1270  real(dp) :: r, r1, r2, c(3), c1(3), c2(3), d, maxc, ssqc
1271  !! At first construct the tree
1272  nclusters = 2*nsph - 1
1273  ! Init the root node
1274  cluster(1, 1) = 1
1275  cluster(2, 1) = nsph
1276  parent(1) = 0
1277  ! Index of the first unassigned node
1278  j = 2
1279  ! Init spheres ordering
1280  do i = 1, nsph
1281  order(i) = i
1282  end do
1283  ! Divide nodes until a single spheres
1284  do i = 1, nclusters
1285  ! The first sphere in i-th node
1286  s = cluster(1, i)
1287  ! The last sphere in i-th node
1288  e = cluster(2, i)
1289  ! Number of spheres in i-th node
1290  n = e - s + 1
1291  ! Divide only if there are 2 or more spheres
1292  if (n .gt. 1) then
1293  ! Use inertial bisection to reorder spheres and cut into 2 halfs
1294  call tree_rib_node_bisect(nsph, csph, n, order(s:e), div)
1295  ! Assign the first half to the j-th node
1296  cluster(1, j) = s
1297  cluster(2, j) = s + div - 1
1298  ! Assign the second half to the (j+1)-th node
1299  cluster(1, j+1) = s + div
1300  cluster(2, j+1) = e
1301  ! Update list of children of i-th node
1302  children(1, i) = j
1303  children(2, i) = j + 1
1304  ! Set parents of new j-th and (j+1)-th node
1305  parent(j) = i
1306  parent(j+1) = i
1307  ! Shift index of the first unassigned node
1308  j = j + 2
1309  ! Set information for a leaf node
1310  else
1311  ! No children nodes
1312  children(:, i) = 0
1313  ! i-th node contains sphere ind(s)
1314  snode(order(s)) = i
1315  end if
1316  end do
1317  !! Compute bounding spheres of each node of the tree
1318  ! Bottom-to-top pass over all nodes of the tree
1319  do i = nclusters, 1, -1
1320  ! In case of a leaf node just use corresponding input sphere as a
1321  ! bounding sphere of the node
1322  if (children(1, i) .eq. 0) then
1323  ! Get correct index of the corresponding input sphere
1324  j = order(cluster(1, i))
1325  ! Get corresponding center and radius
1326  cnode(:, i) = csph(:, j)
1327  rnode(i) = rsph(j)
1328  ! In case of a non-leaf node get minimal sphere that contains bounding
1329  ! spheres of children nodes
1330  else
1331  ! The first child
1332  j = children(1, i)
1333  c1 = cnode(:, j)
1334  r1 = rnode(j)
1335  ! The second child
1336  j = children(2, i)
1337  c2 = cnode(:, j)
1338  r2 = rnode(j)
1339  ! Distance between centers of bounding spheres of children nodes
1340  c = c1 - c2
1341  ! Compute distance by scale and sum of scaled squares
1342  maxc = max(abs(c(1)), abs(c(2)), abs(c(3)))
1343  if (maxc .eq. zero) then
1344  d = zero
1345  else
1346  d = (c(1)/maxc)**2 + (c(2)/maxc)**2 + (c(3)/maxc)**2
1347  d = maxc * sqrt(d)
1348  end if
1349  ! If sphere #2 is completely inside sphere #1 use the first sphere
1350  if (r1 .ge. (r2+d)) then
1351  c = c1
1352  r = r1
1353  ! If sphere #1 is completely inside sphere #2 use the second sphere
1354  else if(r2 .ge. (r1+d)) then
1355  c = c2
1356  r = r2
1357  ! Otherwise use special formula to find a minimal sphere
1358  else
1359  r = (r1+r2+d) / 2
1360  c = c2 + c/d*(r-r2)
1361  end if
1362  ! Assign computed bounding sphere
1363  cnode(:, i) = c
1364  rnode(i) = r
1365  end if
1366  end do
1367 end subroutine tree_rib_build

1378 subroutine tree_rib_node_bisect(nsph, csph, n, order, div)
1379  ! Inputs
1380  integer, intent(in) :: nsph, n
1381  real(dp), intent(in) :: csph(3, nsph)
1382  ! Outputs
1383  integer, intent(inout) :: order(n)
1384  integer, intent(out) :: div
1385  ! Local variables
1386  real(dp) :: c(3), tmp_csph(3, n), s(3)
1387  real(dp), allocatable :: work(:)
1388  external :: dgesvd
1389  integer :: i, l, r, lwork, info, tmp_order(n), istat
1390  ! Get average coordinate
1391  c = zero
1392  do i = 1, n
1393  c = c + csph(:, order(i))
1394  end do
1395  c = c / n
1396  ! Get coordinates minus average in a contiguous array
1397  do i = 1, n
1398  tmp_csph(:, i) = csph(:, order(i)) - c
1399  end do
1400  !! Find right singular vectors
1401  ! Get proper size of temporary workspace
1402  lwork = -1
1403  call dgesvd('N', 'O', 3, n, tmp_csph, 3, s, tmp_csph, 3, tmp_csph, 3, &
1404  & s, lwork, info)
1405  lwork = s(1)
