80 subroutine jacobi_diis(params, constants, workspace, tol, rhs, x, niter, &
81  & x_rel_diff, matvec, dm1vec, norm_func, info)
82  !! Inputs
83  type(ddx_params_type), intent(in) :: params
84  type(ddx_constants_type), intent(in) :: constants
85  real(dp), intent(in) :: tol, rhs(constants % n)
86  !! Temporaries
87  type(ddx_workspace_type), intent(inout) :: workspace
88  !! Outputs
89  real(dp), intent(inout) :: x(constants % n)
90  integer, intent(inout) :: niter
91  real(dp), intent(out) :: x_rel_diff(niter)
92  integer, intent(out) :: info
93  !! External procedures
94  procedure(matvec_interface) :: matvec, dm1vec
95  procedure(norm_interface) :: norm_func
96  !! Local variables
97  integer :: it, nmat
98  real(dp) :: diff, norm, rel_diff
99  !! The code
100  ! DIIS variable
101  nmat = 1
102  ! Iterations
103  do it = 1, niter
104  ! y = rhs - O x
105  call matvec(params, constants, workspace, x, workspace % tmp_y)
106  workspace % tmp_y = rhs - workspace % tmp_y
107  ! x_new = D^-1 y
108  call dm1vec(params, constants, workspace, workspace % tmp_y, &
109  & workspace % tmp_x_new)
110  ! DIIS extrapolation
111  if (params % jacobi_ndiis .gt. 0) then
112  workspace % tmp_x_diis(:, nmat) = workspace % tmp_x_new
113  workspace % tmp_e_diis(:, nmat) = workspace % tmp_x_new - x
114  call diis(constants % n, nmat, params % jacobi_ndiis, &
115  & workspace % tmp_x_diis, workspace % tmp_e_diis, &
116  & workspace % tmp_bmat, workspace % tmp_x_new)
117  end if
118  ! Norm of increment
119  x = workspace % tmp_x_new - x
120  diff = norm_func(params % lmax, constants % nbasis, params % nsph, x)
121  norm = norm_func(params % lmax, constants % nbasis, params % nsph, &
122  & workspace % tmp_x_new)
123 
124  ! compute rel_diff using the rule 0/0 = 0
125  if (diff.eq.zero .and. norm.eq.zero) then
126  rel_diff = zero
127  else
128  rel_diff = diff / norm
129  end if
130 
131  x_rel_diff(it) = rel_diff
132  ! Update solution
133  x = workspace % tmp_x_new
134  ! Check stop condition
135  if (rel_diff .le. tol) then
136  info = 0
137  niter = it
138  return
139  end if
140  end do
141  info = 1
142 endsubroutine jacobi_diis

176  implicit none

178  integer, intent(inout) :: nmat

190  do j = 2, nmat - delta

195  do j = 1, nmat - delta

235 subroutine makeb(n, nmat, ndiis, e, b)
236  implicit none
237  integer, intent(in) :: n, nmat, ndiis
238  real(dp), dimension(n,ndiis), intent(in) :: e
239  real(dp), dimension(ndiis+1,ndiis+1), intent(inout) :: b
240 
241  integer :: i
242  real(dp) :: bij
243  real(dp), parameter :: zero = 0.0d0, one = 1.0d0
244 
245  ! 1st built
246  if (nmat.eq.1) then
247 
248  ! [ 0 | 1 ]
249  ! b = [ --+---- ]
250  ! [ 1 | e*e ]
251 
252  b(1,1) = zero
253  b(1,2) = one
254  b(2,1) = one
255  b(2,2) = dot_product(e(:,1),e(:,1))
256 
257  ! subsequent builts
258  else
259 
260  ! first, update the lagrangian line:
261  b(nmat+1,1) = one
262  b(1,nmat+1) = one
263 
264  ! now, compute the new matrix elements:
265  !$omp parallel do default(none) shared(nmat,e,b) &
266  !$omp private(i,bij) schedule(dynamic)
267  do i = 1, nmat - 1
268  bij = dot_product(e(:,i), e(:,nmat))
269  b(nmat+1, i+1) = bij
270  b(i+1, nmat+1) = bij
271  end do
272  b(nmat+1,nmat+1) = dot_product(e(:,nmat),e(:,nmat))
273  end if
274 end subroutine makeb

277 ! GMRESR algorithm to solve linear system Ax = b
278 !
279 ! author:
280 ! m.botchev, utrecht university, december 1996
281 ! report bugs to botchev@cwi.nl or botchev@excite.com
282 
283 ! Copyright (c) 1996 by M.A.Botchev
284 ! Permission to copy all or part of this work is granted,
285 ! provided that the copies are not made or distributed
286 ! for resale, and that the copyright notice and this
287 ! notice are retained.
288 
289 ! Details to the algorithm may be found in
290 ! H.A. van der Vorst, !. Vuik, "GMRESR: a Family of Nested GMRES
291 ! Methods", Num. Lin. Alg. Appl., vol. 1(4), 369--386 (1994)
292 
293 ! parameter list:
294 ! oktest is TRUE if intermediate residuals should be printed
295 ! n == INTEGER size of problem
296 ! j == INTEGER truncation parameter (work with j last vectors)
297 ! mgmres == INTEGER dimension of the envoked GMRES
298 ! if mgmres.eq.0 then we get GCR algorithm, the simplest
299 ! version of GMRESR
300 ! b == DOUBLE PRECISION righthand side vector
301 ! x == DOUBLE PRECISION initial guess on input,
302 ! (approximate) solution on output
303 ! work == DOUBLE PRECISION work space 1 of size n x (2*j+mgmres+2)
304 ! eps == DOUBLE PRECISION tolerance of stopping criterion.
305 ! process is stopped as soon as
306 ! (1) residual norm has been dumped by factor eps,
307 ! i.e. ||res|| / ||b|| <= eps OR
308 ! (2) maximum number of iterations maxit has been performed
309 ! stc == CHARACTER*3
310 ! Determine stopping criterion (||.|| denotes the 2-norm):
311 ! stc='rel' -- relative stopping crit.: ||res|| < eps*||res0||
312 ! stc='abs' -- absolute stopping crit.: ||res|| < eps
313 ! maxits == INTEGER max. no. outer_iterative_steps/truncation_length on input
314 ! on output it is the actual number of total iterative steps
315 ! resid == DOUBLE PRECISION residual measure (depends on stopping criterion)
316 ! achieved on output
317 ! iflag == INTEGER on output 0 - solution found within tolerance
318 ! 1 - no convergence within niter
319 
320 subroutine gmresr(params, constants, workspace, eps, b, x, &
321  & niter, resid, matvec, info)
322  !! Inputs
323  type(ddx_params_type), intent(in) :: params
324  type(ddx_constants_type), intent(in) :: constants
325  real(dp), intent(in) :: eps, b(constants % n)
326  !! Outputs
327  real(dp), intent(inout) :: x(constants % n)
328  integer, intent(inout) :: niter
329  !real(dp), intent(out) :: x_rel_diff(niter)
330  integer, intent(out) :: info
331  !! Temporary buffers
332  type(ddx_workspace_type), intent(inout) :: workspace
333  !! External procedures
334  procedure(matvec_interface) :: matvec
335  !! Local variables
336 
337 ! list of variables: arrays in alphabetical order,
338 ! then other variables in alphabetical order
339 
340  integer i,its,nits,itsinn,k,n,j
341 
342  double precision alpha, alphai, cknorm, ckres, ddot, dnrm2, &
343  & epsinn, res0,resnor,resid
344 
345 ! distribute work space work(n,*) among some virtual variables;
346 ! namely, we think of columns of work as being occupied by
347 ! c(n,0:j-1), u(n,0:j-1), resid(n), workgmr(n,params % gmresr_dim+1)
348 ! therefore we define "shifts"
349 
350  integer c, u, workres, workgmre
351 ! ----------------------------------------------------------------------------
352 
353  ! Read j and n from the parameters
354  j = params % gmresr_j
355  n = constants % n
356 ! c occupies j columns 0...j-1:
357  c = 0
358 ! u occupies j columns j...2*j-1:
359  u = params % gmresr_j
360 ! resid occupies 1 column No. 2*j:
361  workres = 2*j
362 ! workgmre occupies gmres_dim+1 columns 2*j+1...2*j+gmres_dim+1:
363  workgmre = 2*j+1
364 ! so that we can access, say, to the (k)th column of the virtual
365 ! array c(n,0:j-1) as work(1,c+k),
366 ! virtual residual vector resid(n) is work(1,workres) and so on ...
367 
368 ! ***Furthermore, we build sequences c_k and u_k, k = 0,...,m-1
369 ! but we store only the last j vectors in the following way:
370 ! Assume j=3, then
371 ! --------------------------------------------------------------
372 ! k | number of column of work | columns of work which are vectors
373 ! | in which we store c_k | we actually store and use
374 ! 0 | 0 | c_0 u_0 ...
375 ! 1 | 1 | c_0 c_1 u_0 u_1 ...
376 ! 2 | 2 | c_0 c_1 c_2 u_0 u_1 u_2 ...
377 ! 3 | 0 | c_3 c_1 c_2 u_3 u_1 u_2 ...
378 ! 4 | 1 | c_3 c_4 c_2 u_3 u_4 u_2 ...
379 ! 5 | 2 | c_3 c_4 c_5 u_3 u_4 u_5 ...
380 ! 6 | 0 | c_6 c_4 c_5 u_6 u_4 u_5 ...
381 ! ... | ... | ... ...
382 ! This mapping is performed by function mod(k,j)
383 !
384 
385 ! Reset iteration counter
386  nits= 0
387  its = 0
388 ! Calculate (initial) residual norm
389  call matvec(params, constants, workspace, x, &
390  & workspace % tmp_gmresr(1, workres))
391  alpha = -1
392 !
393  call daxpy(n, alpha, b, 1, workspace % tmp_gmresr(1, workres), 1)
394  call dscal(n, alpha, workspace % tmp_gmresr(1, workres), 1)
395 
396 ! Calculate residual norm and quit if it is zero
397  res0 = dnrm2(n, b, 1)
398  resnor = res0
399  resid = 0
400 
401  if ( res0 .eq. 0.0d0 ) then
402  info = 0
403  niter = 0
404  return
405  end if
406 
407  resid=resnor/res0
408 
409  if ( resid .le. eps ) then
410  info = 0
411  niter = 0
412  return
413  end if
414 
415 ! Main iterative loop ============================
416  k = -1
417  do while (.true.)
418 
419 ! Loop to increment dimension of projection subspace
420  k=k+1
421 
422 ! Number of step (not restart) to be done
423  its = its + 1
424 ! write(*,'(A,i3)') '+++++++++++++++++ its ',its
425 
426 ! - - - - - - - - - - - - - - - - - - - - - - - - -
427 ! This part should deliver
428 ! u(1,k) <-- invA * resid
429 ! where u(1,k) is the k-th column of array u(1:n,0:m) and
430 ! invA is some reasonable approximation to the inverse of A
431 !
432 ! If gmres_dim=0 then no inner iterations are performed
433 ! to get invA, so that invA is just the identity matrix.
434 ! In this case algorithm GMRESR is nothing but GCR
435 !
436 ! Otherwise for inner iterations we perform ONE restart of GMRES
437 ! ATTN: for this implementation it is crucial to perform only
438 ! ONE restart of GMRES
439  if (params % gmresr_dim .eq. 0) then
440 ! u(1,k) := resid
441  call dcopy(n, workspace % tmp_gmresr(1, workres), 1, &
442  & workspace % tmp_gmresr(1, u+mod(k,j)), 1)
443  call matvec(params, constants, workspace, &
444  & workspace % tmp_gmresr(1, u+mod(k,j)), &
445  & workspace % tmp_gmresr(1, c+mod(k,j)))
446  nits=nits+1
447  else
448 ! Solve linear system A*u(1,k)=resid by GMRES
449 ! The stopping criterion for inner iterations is
450 ! always absolute but it is automatically adjusted
451 ! not to be stricter than the stopping criterion for the
452 ! outer iterations. For example, if stop.criterion for
453 ! the outer iterations is relative than absolute stop.
454 ! criterion for the inner iterations is (eps*res0)
455 ! Accuracy for inner iteration:
456 
457  epsinn = eps*res0
458 
459 ! After envoking gmres0 epsinn and itsinn contain actual achieved
460 ! accuracy and number of performed iterations respectively
461 
462  itsinn=params % gmresr_dim
463 
464  call gmres0(params, constants, workspace, n, &
465  & workspace % tmp_gmresr(1, workres), &
466  & workspace % tmp_gmresr(1, u+mod(k,j)), &
467  & workspace % tmp_gmresr(1, c+mod(k,j)), &
468  & workspace % tmp_gmresr(1, workgmre), &
469  & epsinn,itsinn,matvec)
470  !write(*,*) 'GMRES ', work(1,c+mod(k,j))
471 
472  nits=nits+itsinn
473  end if
474 ! - - - - - - - - - - - - - - - - - - - - - - - -
475 
476 ! Inner loop to orthogonalize
477 ! c(1,k) with respect to c(1,k-j),...,c(1,k-1)
478 ! and to update correspondingly
479 ! u(1,k) with respect to u(1,k-j),...,u(1,k-1)
480 ! parameter j is used only here
481  do i = max0(0,k-j),k-1
482  alphai = ddot(n, workspace % tmp_gmresr(1, c+mod(i,j)), 1, &
483  & workspace % tmp_gmresr(1, c+mod(k,j)), 1)
484  call daxpy(n, -alphai, workspace % tmp_gmresr(1, c+mod(i,j)), 1, &
485  & workspace % tmp_gmresr(1, c+mod(k,j)), 1)
486  call daxpy(n, -alphai, workspace % tmp_gmresr(1, u+mod(i,j)), 1, &
487  & workspace % tmp_gmresr(1, u+mod(k,j)), 1)
488  end do
489 
490 ! Normalize c(1,k) and "normalize" u(1,k)
491  cknorm = dnrm2(n, workspace % tmp_gmresr(1, c+mod(k,j)), 1)
492  cknorm = 1 / cknorm
493  call dscal(n, cknorm, workspace % tmp_gmresr(1, c+mod(k,j)), 1)
494  call dscal(n, cknorm, workspace % tmp_gmresr(1, u+mod(k,j)), 1)
495 
496 ! Update current solution and residual
497  ckres = ddot(n, workspace % tmp_gmresr(1, c+mod(k,j)), 1, &
498  & workspace % tmp_gmresr(1, workres), 1)
499  call daxpy(n, ckres, workspace % tmp_gmresr(1, u+mod(k,j)), 1, x, 1)
500  call daxpy(n, -ckres, workspace % tmp_gmresr(1, c+mod(k,j)), 1, &
501  & workspace % tmp_gmresr(1, workres), 1)
502 
503 
504 ! call show(n,10,x,'GMRESR ')
505 
506 ! Calculate residual norm, check convergence
507  resnor = dnrm2(n, workspace % tmp_gmresr(1, workres), 1)
508 
509  resid=resnor/res0
510 
511  if ( resid .le. eps ) then
512  info = 0
513  niter = nits
514  return
515  end if
516  if (its .ge. niter*j) then
517  info = 1
518  niter = nits
519  return
520  end if
521 
522 ! print 11, ' ||res|| = ',resnor
523 ! 11 format(A,d)
524 ! 13 format(i4,A,d)
525 
526  end do
527 ! End of inifinite iterative loop =================
528 ! End of GMRESR subroutine
529  end
530 ! This is the modified GMRES routine gmres0 adapted for GMRESR by
531 ! Mike Botchev, Utrecht University, Dec. 1996
532 ! For detail on how to make GMRES (for GMRESR) cheaper see
533 ! the above-mentioned paper on GMRESR
534 !*************************************************************
535 ! This code was initially written by Youcef Saad
536 ! then revised by Henk A. van der Vorst
537 ! and Mike Botchev (oct. 1996)
538 ! ************************************************************
539 ! gmres algorithm . simple version . (may 23, 1985)
540 ! parameter list:
541 ! oktest == TRUE for printing intermediate results
542 ! n == size of problem
543 ! im == size of krylov subspace: should not exceed 50 in this
544 ! version (can be reset in code. looking at comment below)
545 ! rhs == right hand side
546 ! uu == initial guess for vector u (see above-mentioned paper on GMRESR)
547 ! on input, approximate solution on output
548 ! cc == initial guess for vector c (see above-mentioned paper on GMRESR)
549 ! on input, approximate solution on output
550 ! work0 == work space of size n x (im+1)
551 ! eps == tolerance for stopping criterion. process is stopped
552 ! as soon as ( ||.|| is the euclidean norm):
553 ! || current residual || <= eps
554 ! maxits == maximum number of iterations allowed
555 ! on OUTPUT: actual number of iterations performed
556 ! ----------------------------------------------------------------
557 ! subroutines
558 ! matvec == matrix vector multiplication y <- A*x
559 
560 ! BLAS:
561 ! dcopy == y <-- x routine
562 ! ddot == dot product function
563 ! dnrm2 == euclidean norm function
564 ! daxpy == y <-- y+ax routine
565 ! dscal == x <-- ax routine
566 ! dtsrv == to solve linear system with a triangular matrix
567 !*************************************************************
568 !-------------------------------------------------------------
569 ! arnoldi size should not exceed 10 in this version..
570 ! to reset modify maxdim. BUT: ----------------
571 ! maxdim was set to 10 because of two reasons:
572 ! (1) it is assumed in this implementation that it is cheaper to
573 ! make maxdim vector updates than to make 1 matrix-vector
574 ! multiplication;
575 ! (2) for large maxdim we may lose the orthogonality property
576 ! on which this cheap implementation is based.
577 ! Please keep it in mind changing maxdim
578 !-------------------------------------------------------------
579 !=============================================================================
580 subroutine gmres0(params, constants, workspace, n, rhs, uu, cc, work0, eps, niter, matvec)
581 
582  !! Inputs
583  type(ddx_params_type), intent(in) :: params
584  type(ddx_constants_type), intent(in) :: constants
585  !! Temporary buffers
586  type(ddx_workspace_type), intent(inout) :: workspace
587  integer maxdim,maxd1,md1max
588  parameter(maxdim=20, maxd1=maxdim+1, md1max=maxdim*maxd1)
589  procedure(matvec_interface) :: matvec
590 
591  integer jjj,jj1
592  integer i,i1,its,j,k,k1,niter,n
593  double precision cc(n),coeff,coef1,dabs,ddot,dnrm2,dsqrt,eps,epsmac, &
594  & gam,rhs(n),ro,uu(n),work0(n,params % gmresr_dim+1),t
595 
596  double precision hh(maxd1,maxdim),hh1(maxd1,maxdim),c(maxdim), &
597  & s(maxdim),rs(maxd1),rs1(maxd1)
598

600  & epsmac / 1.d-16 /

602 
603 ! ----------------------------
604 ! Outer loop starts here..
605 ! BUT here (for GMRESR) only 1 outer loop is allowed
606 ! Compute initial residual vector
607 ! ----------------------------
608  10 continue
609 ! do not calculate initial residual first restart because
610 ! initial guess is always zero.
611 ! make initial guess zero:
612  coeff = 0.0
613  call dscal(n,coeff,uu,1)
614 ! make initial residual right-hand side:
615  call dcopy(n,rhs,1,work0,1)
616 
617  ro = dnrm2(n, work0, 1)
618  if ((ro .eq. 0.0d0).or.(ro .le. eps)) then

620  eps = ro
621  niter = its
622  return
623  end if

626  call dscal(n,coeff,work0,1)
627 
628 ! initialize 1-st term of rhs of hessenberg system..
629  rs(1) = ro
630  i = 0
631 
632  4 continue
633  i=i+1
634  its = its + 1
635  i1 = i + 1
636  call matvec(params, constants, workspace, work0(1,i), work0(1,i1))
637 ! -----------------------------------------
638 ! modified gram - schmidt...
639 ! -----------------------------------------
640  do j=1, i
641  t = ddot(n, work0(1,j),1,work0(1,i1),1)
642  hh(j,i) = t
643  call daxpy(n, -t, work0(1,j), 1, work0(1,i1), 1)
644  end do
645  t = dnrm2(n, work0(1,i1), 1)
646  hh(i1,i) = t
647  if (t .ne. 0.0d0)then
648  t = 1 / t
649  call dscal(n, t, work0(1,i1), 1)
650 ! save new column of hh in hh1 to reproduce vector cc later on
651  call dcopy(maxd1,hh(1,i),1,hh1(1,i),1)
652  endif
653 ! done with modified gram schmidt and arnoldi step..
654 
655 ! now update factorization of hh
656  if (i .ne. 1) then
657 ! perform previous transformations on i-th column of h
658  do k=2,i
659  k1 = k-1
660  t = hh(k1,i)
661  hh(k1,i) = c(k1)*t + s(k1)*hh(k,i)
662  hh(k,i) = -s(k1)*t + c(k1)*hh(k,i)
663  end do
664  endif
665  gam = dsqrt(hh(i,i)**2 + hh(i1,i)**2)
666  if (gam .eq. 0.0d0) gam = epsmac
667 
668 ! determine next plane rotation
669  c(i) = hh(i,i)/gam
670  s(i) = hh(i1,i)/gam
671  rs(i1) = -s(i)*rs(i)
672  rs(i) = c(i)*rs(i)
673 
674 ! determine residual norm and test for convergence-
675  hh(i,i) = c(i)*hh(i,i) + s(i)*hh(i1,i)
676  ro = dabs(rs(i1))
677  if ((i .lt. params % gmresr_dim) .and. (ro .gt. eps)) goto 4
678 
679 ! now compute solution. first solve upper triangular system.
680 
681 ! rs := hh(1:i,1:i) ^-1 * rs
682 
683  call dtrsv('U','N','N',i,hh,maxd1,rs,1)
684 ! done with back substitution..
685 
686 ! now form linear combination to get vector uu
687  do j=1, i
688  t = rs(j)
689  call daxpy(n, t, work0(1,j), 1, uu,1)
690  end do
691 ! DO NOT restart outer loop EVEN when necessary (that is crucial
692 ! for this implementation of GMRESR): NEVER goto 10 !
693 ! if (ro .gt. eps .and. its .lt. niter) goto 10
694 
695 ! Finally, reproduce vector cc as cc = A*uu = work0*hh1*rs:
696 ! rs := hh1(1:i1,1:i) * rs
697  coeff = 1
698  coef1 = 0
699  call dgemv('N',i1,i,coeff,hh1,maxd1,rs,1,coef1,rs1,1)
700 
701 ! now multiply Krylov basis vectors work0 by rs:
702 ! cc := work0*rs
703  call dscal(n,coef1,cc,1)
704  do j=1, i1
705  t = rs1(j)
706  call daxpy(n, t, work0(1,j), 1, cc,1)
707  end do
708 
709  niter=its
710  eps=ro
711  return

713  end

716 !
717 ! WARNING: code duplication is bad, but here it makes a very specific sense.
718 !
719 ! in ddLPB, we are solving a linear system with Jacobi DIIS also to apply the
720 ! preconditioner. As calling a solver recursively is a terrible idea, we
721 ! create a copy of jacobi_diis that
722 !
723 ! - takes as an input the size of the system to be solved (no hardcoded dependence
724 ! on nsph)
725 ! - uses copies of the norm subroutine, so that it can also be called with
726 ! a user-given size for arrays
727 !
728 subroutine jacobi_diis_external(params, constants, workspace, n, tol, rhs, x, n_iter, &
729  & matvec, dm1vec, norm_func, info)
730  type(ddx_params_type), intent(in) :: params
731  type(ddx_constants_type), intent(in) :: constants
732  type(ddx_workspace_type), intent(inout) :: workspace
733  ! Inputs
734  integer, intent(in) :: n
735  real(dp), intent(in) :: tol
736  real(dp), dimension(n), intent(in) :: rhs
737  ! Outputs
738  real(dp), dimension(n), intent(inout) :: x
739  integer, intent(inout) :: n_iter, info
740 !
741  external :: matvec, dm1vec
742  procedure(norm_interface) :: norm_func
743  ! Local variables
744  integer :: it, nmat, istatus, lenb, nsph_u
745  real(dp) :: diff, norm, rel_diff
746  logical :: dodiis
747 !
748  real(dp), allocatable :: x_new(:), y(:), x_diis(:,:), e_diis(:,:), bmat(:,:)
749 !
750 !---------------------------------------------------------------------------------------------
751 !
752 !
753 ! DIIS extrapolation flag
754  dodiis = (params%jacobi_ndiis.gt.0)
755 !
756 ! extrapolation required
757  if (dodiis) then
758 !
759 ! allocate workspaces
760  lenb = params % jacobi_ndiis + 1
761  allocate( x_diis(n,params % jacobi_ndiis), e_diis(n,params % jacobi_ndiis), bmat(lenb,lenb) , stat=istatus )
762  if (istatus .ne. 0) then
763  write(*,*) ' jacobi_diis: [1] failed allocation (diis)'
764  stop
765  endif
766 !
767 ! initialize the number of points for diis to one.
768 ! note that nmat is updated by diis.
769  nmat = 1
770 !
771  endif
772 !
773 ! allocate workspaces
774  allocate( x_new(n), y(n) , stat=istatus )
775  if (istatus .ne. 0) then
776  write(*,*) ' jacobi_diis: [2] failed allocation (scratch)'
777  stop
778  endif
779 !
780 ! Jacobi iterations
781 ! =================e
782  do it = 1, n_iter
783 !
784 ! y = rhs - O x
785  call matvec(params, constants, workspace, x, y )
786  y = rhs - y
787 !
788 ! x_new = D^-1 y
789  call dm1vec(params, constants, workspace, y, x_new)
790 !
791 ! DIIS extrapolation
792 ! ==================
793  if (dodiis) then
794 !
795  x_diis(:,nmat) = x_new
796  e_diis(:,nmat) = x_new - x
797 !
798  call diis(n,nmat,params % jacobi_ndiis,x_diis,e_diis,bmat,x_new)
799 !
800  endif
801 !
802 ! increment
803  x = x_new - x
804 !
805 ! warning: in the following, we use a quick and dirty hack to pass
806 ! to the norm function arrays double the size than in non ddlpb calculations
807 ! by defining a fake number of spheres with is n/nbasis (i.e., 2*nsph in
808 ! ddlpb).
809 !
810  nsph_u = n/constants % nbasis
811  diff = norm_func(params % lmax, constants % nbasis, nsph_u, x)
812  norm = norm_func(params % lmax, constants % nbasis, nsph_u, x_new)
813  ! compute rel_diff using the rule 0/0 = 0
814  if (diff.eq.zero .and. norm.eq.zero) then
815  rel_diff = zero
816  else
817  rel_diff = diff / norm
818  end if

820 ! update
821  x = x_new
822 !
823 ! EXIT Jacobi loop here
824 ! =====================
825  if (rel_diff .le. tol) then
826  info = 0
827  n_iter = it
828  return
829  end if
830 !
831  enddo
832 !
833 ! something went wrong...
834  info = 1
835 !
836  return
