459  ! Outputs

492  end if

499  end if

502  if(ngrid .lt. 0) then
503  write(*, "(3A)") "Error on the 5th line of a config file ", fname, &
504  & ": `ngrid` must be a non-negative integer value."
505  stop 1
506  end if
507  ! Dielectric permittivity constant of the solvent
508  read(100, *) eps
509  if(eps .lt. zero) then
510  write(*, "(3A)") "Error on the 6th line of a config file ", fname, &
511  & ": `eps` must be a non-negative floating point value."
512  stop 1
513  end if
514  ! Shift of the regularized characteristic function
515  read(100, *) se
516  if((se .lt. -one) .or. (se .gt. one)) then
517  write(*, "(3A)") "Error on the 7th line of a config file ", fname, &
518  & ": `se` must be a floating point value in a range [-1, 1]."

533  stop 1
534  end if

536  read(100, *) itersolver
537  if((itersolver .lt. 1) .or. (itersolver .gt. 2)) then

541  end if

548  end if
549  ! Maximum number of iterations for the iterative solver

562  end if

567  write(*, "(3A)") "Error on the 14th line of a config file ", fname, &

569  stop 1
570  end if

574  read(100, *) gmresr_dim
575  if((gmresr_dim .lt. 0)) then
576  write(*, "(3A)") "Error on the 15th line of a config file ", fname, &
577  & ": `gmresr_dim` must be a non-negative integer value."
578  stop 1
579  end if
580  ! Dimension of the envoked GMRESR. In case of 0 GMRESR becomes the GCR
581  ! solver, one of the simplest versions of GMRESR. Referenced only if
582  ! GMRESR solver is used.
583  ! Whether to compute (1) or not (0) forces as analytical gradients
584  read(100, *) force

586  write(*, "(3A)") "Error on the 16th line of a config file ", fname, &
587  & ": `force` must be an integer value of a value 0 or 1."
588  stop 1
589  end if
590  ! Whether to use (1) or not (0) the FMM to accelerate computations
591  read(100, *) fmm
592  if((fmm .lt. 0) .or. (fmm .gt. 1)) then
593  write(*, "(3A)") "Error on the 17th line of a config file ", fname, &
594  & ": `fmm` must be an integer value of a value 0 or 1."
595  stop 1
596  end if
597  ! Max degree of multipole spherical harmonics for the FMM
598  read(100, *) pm

600  write(*, "(3A)") "Error on the 18th line of a config file ", fname, &
601  & ": `pm` must be a non-negative integer value."
602  stop 1
603  end if
604  ! Max degree of local spherical harmonics for the FMM
605  read(100, *) pl

607  write(*, "(3A)") "Error on the 19th line of a config file ", fname, &
608  & ": `pl` must be a non-negative integer value."
609  stop 1
610  end if
611  ! Number of input spheres
612  read(100, *) nsph
613  if(nsph .le. 0) then
614  write(*, "(3A)") "Error on the 20th line of a config file ", fname, &
615  & ": `nsph` must be a positive integer value."
616  stop 1
617  end if
618  ! Coordinates, radii and charges
619  allocate(charge(nsph), x(nsph), y(nsph), z(nsph), rvdw(nsph), stat=istatus)
620  if(istatus .ne. 0) then
621  write(*, "(2A)") "Could not allocate space for coordinates, radii ", &
622  & "and charges of atoms"
623  stop 1
624  end if
625  do i = 1, nsph
626  read(100, *) charge(i), x(i), y(i), z(i), rvdw(i)
627  end do
628  ! Finish reading
629  close(100)
630  !! Convert Angstrom input into Bohr units

632  y = y * tobohr
633  z = z * tobohr
634  rvdw = rvdw * tobohr
635 
636  ! adjust ngrid
637  call closest_supported_lebedev_grid(ngrid)
638 
639  !! Initialize ddx_data object
640  call ddinit(nsph, charge, x, y, z, rvdw, model, lmax, ngrid, force, fmm, &
641  & pm, pl, se, eta, eps, kappa, itersolver, &
642  & maxiter, jacobi_ndiis, gmresr_j, gmresr_dim, nproc, ddx_data, info)
643  !! Clean local temporary data
644  deallocate(charge, x, y, z, rvdw, stat=istatus)
645  if(istatus .ne. 0) then
646  write(*, "(2A)") "Could not deallocate space for coordinates, ", &
647  & "radii and charges of atoms"
648  stop 1
649  end if

874 subroutine ylmbas2(x, sphcoo, p, vscales, vylm, vplm, vcos, vsin)
875  ! Inputs
876  integer, intent(in) :: p
877  real(dp), intent(in) :: x(3)
878  real(dp), intent(in) :: vscales((p+1)**2)
879  ! Outputs
880  real(dp), intent(out) :: sphcoo(5)
881  real(dp), intent(out) :: vylm((p+1)**2), vplm((p+1)**2)
882  real(dp), intent(out) :: vcos(p+1), vsin(p+1)
883  ! Local variables
884  integer :: l, m, ind
885  real(dp) :: max12, ssq12, tmp, rho, ctheta, stheta, cphi, sphi
886  ! Get rho cos(theta), sin(theta), cos(phi) and sin(phi) from the cartesian
887  ! coordinates of x. To support full range of inputs we do it via a scale
888  ! and a sum of squares technique.
889  ! At first we compute x(1)**2 + x(2)**2
890  if (x(1) .eq. zero) then
891  max12 = abs(x(2))
892  ssq12 = one
893  else if (abs(x(2)) .gt. abs(x(1))) then
894  max12 = abs(x(2))
895  ssq12 = one + (x(1)/x(2))**2
896  else
897  max12 = abs(x(1))
898  ssq12 = one + (x(2)/x(1))**2
899  end if
900  ! Then we compute rho
901  if (x(3) .eq. zero) then
902  rho = max12 * sqrt(ssq12)
903  else if (abs(x(3)) .gt. max12) then
904  rho = one + ssq12*(max12/x(3))**2
905  rho = abs(x(3)) * sqrt(rho)
906  else
907  rho = ssq12 + (x(3)/max12)**2
908  rho = max12 * sqrt(rho)
909  end if
910  ! In case x=0 just exit without setting any other variable
911  if (rho .eq. zero) then
912  sphcoo = zero
913  return
914  end if
915  ! Length of a vector x(1:2)
916  stheta = max12 * sqrt(ssq12)
917  ! Case x(1:2) != 0
918  if (stheta .ne. zero) then
919  ! Evaluate cos(m*phi) and sin(m*phi) arrays
920  cphi = x(1) / stheta
921  sphi = x(2) / stheta
922  call trgev(cphi, sphi, p, vcos, vsin)
923  ! Normalize ctheta and stheta
924  ctheta = x(3) / rho
925  stheta = stheta / rho
926  ! Evaluate associated Legendre polynomials
927  call polleg(ctheta, stheta, p, vplm)
928  ! Construct spherical harmonics
929  do l = 0, p
930  ! Offset of a Y_l^0 harmonic in vplm and vylm arrays
931  ind = l**2 + l + 1
932  ! m = 0 implicitly uses `vcos(1) = 1`
933  vylm(ind) = vscales(ind) * vplm(ind)
934  do m = 1, l
935  ! only P_l^m for non-negative m is used/defined
936  tmp = vplm(ind+m) * vscales(ind+m)
937  ! m > 0
938  vylm(ind+m) = tmp * vcos(m+1)
939  ! m < 0
940  vylm(ind-m) = tmp * vsin(m+1)
941  end do
942  end do
943  ! Case of x(1:2) = 0 and x(3) != 0
944  else
945  ! Set spherical coordinates
946  cphi = one
947  sphi = zero
948  ctheta = sign(one, x(3))
949  stheta = zero
950  ! Set output arrays vcos and vsin
951  vcos = one
952  vsin = zero
953  ! Evaluate spherical harmonics. P_l^m = 0 for m > 0. In the case m = 0
954  ! it depends if l is odd or even. Additionally, vcos = one and vsin =
955  ! zero for all elements
956  vylm = zero
957  vplm = zero
958  do l = 0, p, 2
959  ind = l**2 + l + 1
960  ! only case m = 0
961  vplm(ind) = one
962  vylm(ind) = vscales(ind)
963  end do
964  do l = 1, p, 2
965  ind = l**2 + l + 1
966  ! only case m = 0
967  vplm(ind) = ctheta
968  vylm(ind) = ctheta * vscales(ind)
969  end do
970  end if
971  ! Set output spherical coordinates
972  sphcoo(1) = rho
973  sphcoo(2) = ctheta
974  sphcoo(3) = stheta
975  sphcoo(4) = cphi
976  sphcoo(5) = sphi
977 end subroutine ylmbas2
