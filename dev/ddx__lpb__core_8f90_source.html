264 subroutine adjrhs_lpb(params, constants, workspace, isph, xi, vlm, basloc, &

267  !! input/output
268  type(ddx_params_type), intent(in) :: params
269  type(ddx_constants_type), intent(in) :: constants

271  real(dp), dimension((params % lmax+1)**2), intent(out) :: vlm
272  real(dp), dimension((params % lmax+1)**2), intent(out) :: basloc, vplm
273  real(dp), dimension(params % lmax+1), intent(out) :: vcos, vsin

283  !loop over neighbors of i-sphere
284  do ij = constants % inl(isph), constants % inl(isph+1)-1
285  !j-sphere is neighbor
286  jsph = constants % nl(ij)
287  !loop over integration points

291  & constants % cgrid(:,ig) - params % csph(:,isph)

301  & vplm, vcos, vsin)

306  !compute W_n^ji

308  oji = xji/ constants % fi(ig,jsph)
309  else
310  oji = xji
311  endif

314  !loop over l

321  enddo
322  endif
323  enddo
324  enddo

379  !! input/output
380  type(ddx_params_type), intent(in) :: params
381  type(ddx_constants_type), intent(in) :: constants
382  type(ddx_workspace_type), intent(inout) :: workspace
383  integer, intent(in) :: isph
384  real(dp), dimension(constants % nbasis, params % nsph), intent(in) :: Xe
385  real(dp), dimension(params % ngrid), intent(in) :: Xadj_e
386  real(dp), dimension(constants % nbasis), intent(inout) :: basloc, vplm
387  real(dp), dimension(3, constants % nbasis), intent(inout) :: dbasloc
388  real(dp), dimension(params % lmax+1), intent(inout) :: vcos, vsin
389  real(dp), dimension(3), intent(inout) :: force_e

391  ! Local Variables
392  integer :: igrid, ineigh, jsph, l, ind, m
393  real(dp), dimension(0:params % lmax) :: SI_rijn
394  real(dp), dimension(0:params % lmax) :: DI_rijn
395  ! beta : Eq.(53) Stamm.etal.18
396  ! tlow : Lower bound for switch region
397  ! thigh : Upper bound for switch region
398  ! f1 : First factor in alpha computation
399  ! f2 : Second factor in alpha computation
400  real(dp) :: rijn, tij, beta, tlow, thigh, xij, oij, f1, f2, f3
401  ! alpha : Eq.(52) Stamm.etal.18
402  ! va : Eq.(54) Stamm.etal.18
403  real(dp) :: vij(3), sij(3), alpha(3), va(3), rj, vtij(3)
404  real(dp), external :: dnrm2
405  real(dp) :: work(params % lmax+1)
406  complex(dp) :: work_complex(params % lmax+1)
407 
408  si_rijn = 0
409  di_rijn = 0
410 
411  tlow = one - pt5*(one - params % se)*params % eta
412  thigh = one + pt5*(one + params % se)*params % eta
413 
414  ! Loop over grid points
415  do igrid = 1, params % ngrid
416  va = zero
417  do ineigh = constants % inl(isph), constants % inl(isph+1) - 1
418  jsph = constants % nl(ineigh)
419  vij = params % csph(:,isph) + &
420  & params % rsph(isph)*constants % cgrid(:,igrid) - &
421  & params % csph(:,jsph)
422  rijn = dnrm2(3, vij, 1)
423  tij = rijn/params % rsph(jsph)
424  rj = params % rsph(jsph)
425  if (tij.ge.thigh) cycle
426  ! Computation of modified spherical Bessel function values
427  sij = vij/rijn
428  vtij = vij*params % kappa
429  call fmm_l2p_bessel_grad(vtij, params % rsph(jsph)*params % kappa, &
430  & params % lmax, constants % vscales, params % kappa, &
431  & xe(:, jsph), zero, alpha)
432  call fmm_l2p_bessel_work(vtij, params % lmax, constants % vscales, &
433  & constants % SI_ri(:, jsph), one, xe(:, jsph), zero, beta, &
434  & work_complex, work)
435  xij = fsw(tij, params % se, params % eta)
436  if (constants % fi(igrid,isph).gt.one) then
437  oij = xij/constants % fi(igrid,isph)
438  f2 = -oij/constants % fi(igrid,isph)
439  else
440  oij = xij
441  f2 = zero
442  end if
443  f1 = oij
444  va(:) = va(:) + f1*alpha(:) + beta*f2*constants % zi(:,igrid,isph)
445  if (tij .gt. tlow) then
446  f3 = beta*dfsw(tij,params % se,params % eta)/params % rsph(jsph)
447  if (constants % fi(igrid,isph).gt.one) &
448  & f3 = f3/constants % fi(igrid,isph)
449  va(:) = va(:) + f3*sij(:)
450  end if
451  end do
452  force_e = force_e - constants % wgrid(igrid)*xadj_e(igrid)*va(:)
453  end do
454 end subroutine fdoka_b_xe
