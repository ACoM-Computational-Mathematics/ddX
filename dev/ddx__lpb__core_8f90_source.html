286 subroutine adjrhs_lpb(params, constants, workspace, isph, xi, vlm, basloc, &
287  & vplm, vcos, vsin, tmp_bessel)
288  implicit none
289  !! input/output
290  type(ddx_params_type), intent(in) :: params
291  type(ddx_constants_type), intent(in) :: constants
292  real(dp), dimension(params % ngrid, params % nsph), intent(in) :: xi
293  real(dp), dimension((params % lmax+1)**2), intent(out) :: vlm
294  real(dp), dimension((params % lmax+1)**2), intent(out) :: basloc, vplm
295  real(dp), dimension(params % lmax+1), intent(out) :: vcos, vsin
296  complex(dp), dimension(max(2, params % lmax+1)), intent(out) :: tmp_bessel
297  type(ddx_workspace_type), intent(inout) :: workspace
298  integer, intent(in) :: isph
299  !! local
300  integer :: ij, jsph, ig, l, ind, m
301  real(dp) :: vji(3), vvji, tji, sji(3), xji, oji, fac
302  real(dp) :: rho, ctheta, stheta, cphi, sphi
303  real(dp), dimension(constants % nbasis) :: fac_hsp
304 
305  !loop over neighbors of i-sphere
306  do ij = constants % inl(isph), constants % inl(isph+1)-1
307  !j-sphere is neighbor
308  jsph = constants % nl(ij)
309  !loop over integration points
310  do ig = 1, params % ngrid
311  !compute t_n^ji = | r_j + \rho_j s_n - r_i | / \rho_i
312  vji = params % csph(:,jsph) + params % rsph(jsph)* &
313  & constants % cgrid(:,ig) - params % csph(:,isph)
314  vvji = sqrt(dot_product(vji,vji))
315  tji = vvji/params % rsph(isph)
316  !point is INSIDE i-sphere (+ transition layer)
317  if ( tji.lt.( one + (params % se+one)/two*params % eta ) ) then
318  !compute s_n^ji
319  sji = vji/vvji
320  call ylmbas(sji, rho, ctheta, stheta, cphi, &
321  & sphi, params % lmax, &
322  & constants % vscales, basloc, &
323  & vplm, vcos, vsin)
324  call inthsp_adj(params, constants, vvji, params % rsph(isph), isph, &
325  & basloc, fac_hsp, tmp_bessel)
326  !compute \chi( t_n^ji )
327  xji = fsw( tji, params % se, params % eta )
328  !compute W_n^ji
329  if ( constants % fi(ig,jsph).gt.one ) then
330  oji = xji/ constants % fi(ig,jsph)
331  else
332  oji = xji
333  endif
334  !compute w_n * xi(n,j) * W_n^ji
335  fac = constants % wgrid(ig) * xi(ig,jsph) * oji
336  !loop over l
337  do l = 0, params % lmax
338  ind = l*l + l + 1
339  !loop over m
340  do m = -l,l
341  vlm(ind+m) = vlm(ind+m) + fac*fac_hsp(ind+m)
342  enddo
343  enddo
344  endif
345  enddo
346  enddo
347 end subroutine adjrhs_lpb
